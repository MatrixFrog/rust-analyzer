1//! Many kinds of items or constructs can have generic parameters: functions,
2//! structs, impls, traits, etc. This module provides a common HIR for these
3//! generic parameters. See also the `Generics` type and the `generics_of` query
4//! in rustc.
5
6use std::sync::LazyLock;

8use either::Either;
9use hir_expand::name::{AsName, Name};

12use syntax::ast::{self, HasName, HasTypeBounds};
13use thin_vec::ThinVec;

26pub(crate) struct GenericParamsCollector<'db, 'c> {
27    expr_collector: &'c mut ExprCollector<'db>,
28    type_or_consts: Arena<TypeOrConstParamData>,
29    lifetimes: Arena<LifetimeParamData>,
30    where_predicates: Vec<WherePredicate>,
31    parent: GenericDefId,
32}
33
34impl<'db, 'c> GenericParamsCollector<'db, 'c> {
35    pub(crate) fn new(expr_collector: &'c mut ExprCollector<'db>, parent: GenericDefId) -> Self {
36        Self {
37            expr_collector,
38            type_or_consts: Default::default(),
39            lifetimes: Default::default(),
40            where_predicates: Default::default(),
41            parent,
42        }
43    }
44
45    pub(crate) fn fill_self_param(&mut self, bounds: Option<ast::TypeBoundList>) {
46        let self_ = Name::new_symbol_root(sym::Self_.clone());
47        let idx = self.type_or_consts.alloc(
48            TypeParamData {
49                name: Some(self_.clone()),
50                default: None,
51                provenance: TypeParamProvenance::TraitSelf,
52            }
53            .into(),
54        );
55        let type_ref = TypeRef::TypeParam(TypeParamId::from_unchecked(TypeOrConstParamId {
56            parent: self.parent,
57            local_id: idx,
58        }));
59        let self_ = self.expr_collector.alloc_type_ref_desugared(type_ref);
60        if let Some(bounds) = bounds {
61            self.lower_bounds(Some(bounds), Either::Left(self_));
62        }
63    }
64
65    pub(crate) fn lower(
66        &mut self,
67        generic_param_list: Option<ast::GenericParamList>,
68        where_clause: Option<ast::WhereClause>,
69    ) {
70        if let Some(params) = generic_param_list {
71            self.lower_param_list(params)
72        }
73        if let Some(where_clause) = where_clause {
74            self.lower_where_predicates(where_clause);
75        }
76    }
77
78    fn lower_param_list(&mut self, params: ast::GenericParamList) {
79        for generic_param in params.generic_params() {
80            let enabled = self.expr_collector.expander.is_cfg_enabled(
81                self.expr_collector.db,
82                self.expr_collector.module.krate(),
83                &generic_param,
84            );
85            if !enabled {
86                continue;
87            }
88
89            match generic_param {
90                ast::GenericParam::TypeParam(type_param) => {
91                    let name = type_param.name().map_or_else(Name::missing, |it| it.as_name());
92                    let default = type_param
93                        .default_type()
94                        .map(|it| self.expr_collector.lower_type_ref(it, &mut |_| TypeRef::Error));
95                    let param = TypeParamData {
96                        name: Some(name.clone()),
97                        default,
98                        provenance: TypeParamProvenance::TypeParamList,
99                    };
100                    let idx = self.type_or_consts.alloc(param.into());
101                    let type_ref =
102                        TypeRef::TypeParam(TypeParamId::from_unchecked(TypeOrConstParamId {
103                            parent: self.parent,
104                            local_id: idx,
105                        }));
106                    let type_ref = self.expr_collector.alloc_type_ref_desugared(type_ref);
107                    self.lower_bounds(type_param.type_bound_list(), Either::Left(type_ref));
108                }
109                ast::GenericParam::ConstParam(const_param) => {
110                    let name = const_param.name().map_or_else(Name::missing, |it| it.as_name());
111                    let ty = self
112                        .expr_collector
113                        .lower_type_ref_opt(const_param.ty(), &mut |_| TypeRef::Error);
114                    let param = ConstParamData {
115                        name,
116                        ty,
117                        default: const_param
118                            .default_val()
119                            .map(|it| self.expr_collector.lower_const_arg(it)),
120                    };
121                    let _idx = self.type_or_consts.alloc(param.into());
122                }
123                ast::GenericParam::LifetimeParam(lifetime_param) => {
124                    let lifetime_ref =
125                        self.expr_collector.lower_lifetime_ref_opt(lifetime_param.lifetime());
126                    if let LifetimeRef::Named(name) = &lifetime_ref {
127                        let param = LifetimeParamData { name: name.clone() };
128                        let _idx = self.lifetimes.alloc(param);
129                        self.lower_bounds(
130                            lifetime_param.type_bound_list(),
131                            Either::Right(lifetime_ref),
132                        );
133                    }
134                }
135            }
136        }
137    }
138
139    fn lower_where_predicates(&mut self, where_clause: ast::WhereClause) {
140        for pred in where_clause.predicates() {
141            let target = if let Some(type_ref) = pred.ty() {
142                Either::Left(self.expr_collector.lower_type_ref(type_ref, &mut |_| TypeRef::Error))
143            } else if let Some(lifetime) = pred.lifetime() {
144                Either::Right(self.expr_collector.lower_lifetime_ref(lifetime))
145            } else {
146                continue;
147            };
148
149            let lifetimes: Option<Box<_>> = pred.generic_param_list().map(|param_list| {
150                // Higher-Ranked Trait Bounds
151                param_list
152                    .lifetime_params()
153                    .map(|lifetime_param| {
154                        lifetime_param
155                            .lifetime()
156                            .map_or_else(Name::missing, |lt| Name::new_lifetime(&lt.text()))
157                    })
158                    .collect()
159            });
160            for bound in pred.type_bound_list().iter().flat_map(|l| l.bounds()) {
161                self.lower_type_bound_as_predicate(bound, lifetimes.as_deref(), target.clone());
162            }
163        }
164    }
165
166    fn lower_bounds(
167        &mut self,
168        type_bounds: Option<ast::TypeBoundList>,
169        target: Either<TypeRefId, LifetimeRef>,
170    ) {
171        for bound in type_bounds.iter().flat_map(|type_bound_list| type_bound_list.bounds()) {
172            self.lower_type_bound_as_predicate(bound, None, target.clone());
173        }
174    }
175
176    fn lower_type_bound_as_predicate(
177        &mut self,
178        bound: ast::TypeBound,
179        hrtb_lifetimes: Option<&[Name]>,
180        target: Either<TypeRefId, LifetimeRef>,
181    ) {
182        let bound = self.expr_collector.lower_type_bound(
183            bound,
184            &mut Self::lower_argument_impl_trait(
185                &mut self.type_or_consts,
186                &mut self.where_predicates,
187                self.parent,
188            ),
189        );
190        let predicate = match (target, bound) {
191            (_, TypeBound::Error | TypeBound::Use(_)) => return,
192            (Either::Left(type_ref), bound) => match hrtb_lifetimes {
193                Some(hrtb_lifetimes) => WherePredicate::ForLifetime {
194                    lifetimes: hrtb_lifetimes.to_vec().into_boxed_slice(),
195                    target: WherePredicateTypeTarget::TypeRef(type_ref),
196                    bound,
197                },
198                None => WherePredicate::TypeBound {
199                    target: WherePredicateTypeTarget::TypeRef(type_ref),
200                    bound,
201                },
202            },
203            (Either::Right(lifetime), TypeBound::Lifetime(bound)) => {
204                WherePredicate::Lifetime { target: lifetime, bound }
205            }
206            (Either::Right(_), TypeBound::ForLifetime(..) | TypeBound::Path(..)) => return,
207        };
208        self.where_predicates.push(predicate);
209    }
210
211    pub(crate) fn collect_impl_trait<R>(
212        &mut self,
213        cb: impl FnOnce(&mut ExprCollector<'_>, &mut dyn FnMut(ThinVec<TypeBound>) -> TypeRef) -> R,
214    ) -> R {
215        cb(
216            self.expr_collector,
217            &mut Self::lower_argument_impl_trait(
218                &mut self.type_or_consts,
219                &mut self.where_predicates,
220                self.parent,
221            ),
222        )
223    }
224
225    fn lower_argument_impl_trait(
226        type_or_consts: &mut Arena<TypeOrConstParamData>,
227        where_predicates: &mut Vec<WherePredicate>,
228        parent: GenericDefId,
229    ) -> impl FnMut(ThinVec<TypeBound>) -> TypeRef {
230        move |impl_trait_bounds| {
231            let param = TypeParamData {
232                name: None,
233                default: None,
234                provenance: TypeParamProvenance::ArgumentImplTrait,
235            };
236            let param_id = type_or_consts.alloc(param.into());
237            for bound in impl_trait_bounds {
238                where_predicates.push(WherePredicate::TypeBound {
239                    target: WherePredicateTypeTarget::TypeOrConstParam(param_id),
240                    bound: bound.clone(),
241                });
242            }
243            TypeRef::TypeParam(TypeParamId::from_unchecked(TypeOrConstParamId {
244                parent,
245                local_id: param_id,
246            }))
247        }
248    }
249
250    pub(crate) fn finish(self) -> Arc<GenericParams> {
251        let Self {
252            mut lifetimes,
253            mut type_or_consts,
254            mut where_predicates,
255            expr_collector: _,
256            parent: _,
257        } = self;
258
259        if lifetimes.is_empty() && type_or_consts.is_empty() && where_predicates.is_empty() {
260            static EMPTY: LazyLock<Arc<GenericParams>> = LazyLock::new(|| {
261                Arc::new(GenericParams {
262                    lifetimes: Arena::new(),
263                    type_or_consts: Arena::new(),
264                    where_predicates: Box::default(),
265                })
266            });
267            return Arc::clone(&EMPTY);
268        }
269
270        lifetimes.shrink_to_fit();
271        type_or_consts.shrink_to_fit();
272        where_predicates.shrink_to_fit();
273        Arc::new(GenericParams {
274            type_or_consts,
275            lifetimes,
276            where_predicates: where_predicates.into_boxed_slice(),
277        })
278    }
