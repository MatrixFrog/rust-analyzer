58    pub(super) fn lower_module_items(mut self, item_owner: &dyn HasModuleItem) -> ItemTree {

64    pub(super) fn lower_macro_stmts(mut self, stmts: ast::MacroStmts) -> ItemTree {
65        self.tree.top_level = stmts
66            .statements()
67            .filter_map(|stmt| {
68                match stmt {
69                    ast::Stmt::Item(item) => Some(item),
70                    // Macro calls can be both items and expressions. The syntax library always treats
71                    // them as expressions here, so we undo that.
72                    ast::Stmt::ExprStmt(es) => match es.expr()? {
73                        ast::Expr::MacroExpr(expr) => {
74                            cov_mark::hit!(macro_call_in_macro_stmts_is_added_to_item_tree);
75                            Some(expr.macro_call()?.into())
76                        }
77                        _ => None,
78                    },
79                    _ => None,
80                }
81            })
82            .flat_map(|item| self.lower_mod_item(&item))
83            .collect();
84
85        if let Some(ast::Expr::MacroExpr(tail_macro)) = stmts.expr() {
86            if let Some(call) = tail_macro.macro_call() {
87                cov_mark::hit!(macro_stmt_with_trailing_macro_expr);
88                if let Some(mod_item) = self.lower_mod_item(&call.into()) {
89                    self.tree.top_level.push(mod_item);
90                }
91            }
92        }
93
94        self.tree
95    }

97    pub(super) fn lower_block(mut self, block: &ast::BlockExpr) -> ItemTree {
98        self.tree.attrs.insert(AttrOwner::TopLevel, RawAttrs::new(self.db, block, self.span_map()));
99        self.tree.top_level = block
100            .statements()
101            .filter_map(|stmt| match stmt {
102                ast::Stmt::Item(item) => self.lower_mod_item(&item),
103                // Macro calls can be both items and expressions. The syntax library always treats
104                // them as expressions here, so we undo that.
105                ast::Stmt::ExprStmt(es) => match es.expr()? {
106                    ast::Expr::MacroExpr(expr) => self.lower_mod_item(&expr.macro_call()?.into()),
107                    _ => None,
108                },
109                _ => None,
110            })
111            .collect();
112        if let Some(ast::Expr::MacroExpr(expr)) = block.tail_expr() {
113            if let Some(call) = expr.macro_call() {
114                if let Some(mod_item) = self.lower_mod_item(&call.into()) {
115                    self.tree.top_level.push(mod_item);
116                }
117            }
118        }
119
120        self.tree
121    }

127    fn lower_mod_item(&mut self, item: &ast::Item) -> Option<ModItem> {
128        let mod_item: ModItem = match item {
129            ast::Item::Struct(ast) => self.lower_struct(ast)?.into(),
130            ast::Item::Union(ast) => self.lower_union(ast)?.into(),
131            ast::Item::Enum(ast) => self.lower_enum(ast)?.into(),
132            ast::Item::Fn(ast) => self.lower_function(ast)?.into(),
133            ast::Item::TypeAlias(ast) => self.lower_type_alias(ast)?.into(),
134            ast::Item::Static(ast) => self.lower_static(ast)?.into(),
135            ast::Item::Const(ast) => self.lower_const(ast).into(),
136            ast::Item::Module(ast) => self.lower_module(ast)?.into(),
137            ast::Item::Trait(ast) => self.lower_trait(ast)?.into(),
138            ast::Item::TraitAlias(ast) => self.lower_trait_alias(ast)?.into(),
139            ast::Item::Impl(ast) => self.lower_impl(ast).into(),
140            ast::Item::Use(ast) => self.lower_use(ast)?.into(),
141            ast::Item::ExternCrate(ast) => self.lower_extern_crate(ast)?.into(),
142            ast::Item::MacroCall(ast) => self.lower_macro_call(ast)?.into(),
143            ast::Item::MacroRules(ast) => self.lower_macro_rules(ast)?.into(),
144            ast::Item::MacroDef(ast) => self.lower_macro_def(ast)?.into(),
145            ast::Item::ExternBlock(ast) => self.lower_extern_block(ast).into(),
146        };
147        let attrs = RawAttrs::new(self.db, item, self.span_map());
148        self.add_attrs(mod_item.into(), attrs);
149
150        Some(mod_item)
151    }

153    fn add_attrs(&mut self, item: AttrOwner, attrs: RawAttrs) {
154        if !attrs.is_empty() {
155            match self.tree.attrs.entry(item) {
156                Entry::Occupied(mut entry) => {
157                    *entry.get_mut() = entry.get().merge(attrs);
158                }
159                Entry::Vacant(entry) => {
160                    entry.insert(attrs);
161                }
162            }
163        }
164    }

166    fn lower_assoc_item(&mut self, item_node: &ast::AssocItem) -> Option<AssocItem> {
167        let item: AssocItem = match item_node {
168            ast::AssocItem::Fn(ast) => self.lower_function(ast).map(Into::into),
169            ast::AssocItem::TypeAlias(ast) => self.lower_type_alias(ast).map(Into::into),
170            ast::AssocItem::Const(ast) => Some(self.lower_const(ast).into()),
171            ast::AssocItem::MacroCall(ast) => self.lower_macro_call(ast).map(Into::into),
172        }?;
173        let attrs = RawAttrs::new(self.db, item_node, self.span_map());
174        self.add_attrs(
175            match item {
176                AssocItem::Function(it) => AttrOwner::ModItem(ModItem::Function(it)),
177                AssocItem::TypeAlias(it) => AttrOwner::ModItem(ModItem::TypeAlias(it)),
178                AssocItem::Const(it) => AttrOwner::ModItem(ModItem::Const(it)),
179                AssocItem::MacroCall(it) => AttrOwner::ModItem(ModItem::MacroCall(it)),
180            },
181            attrs,
182        );
183        Some(item)
184    }

186    fn lower_struct(&mut self, strukt: &ast::Struct) -> Option<FileItemTreeId<Struct>> {
187        let visibility = self.lower_visibility(strukt);
188        let name = strukt.name()?.as_name();
189        let ast_id = self.source_ast_id_map.ast_id(strukt);
190        let (fields, kind, attrs) = self.lower_fields(&strukt.kind());
191        let res = Struct { name, visibility, fields, shape: kind, ast_id };
192        let id = id(self.data().structs.alloc(res));
193
194        for (idx, attr) in attrs {
195            self.add_attrs(
196                AttrOwner::Field(
197                    FieldParent::Struct(id),
198                    Idx::from_raw(RawIdx::from_u32(idx as u32)),
199                ),
200                attr,
201            );
202        }
203        Some(id)
204    }

206    fn lower_fields(
207        &mut self,
208        strukt_kind: &ast::StructKind,
209    ) -> (Box<[Field]>, FieldsShape, Vec<(usize, RawAttrs)>) {
210        match strukt_kind {
211            ast::StructKind::Record(it) => {
212                let mut fields = vec![];
213                let mut attrs = vec![];
214
215                for (i, field) in it.fields().enumerate() {
216                    let data = self.lower_record_field(&field);
217                    fields.push(data);
218                    let attr = RawAttrs::new(self.db, &field, self.span_map());
219                    if !attr.is_empty() {
220                        attrs.push((i, attr))
221                    }
222                }
223                (fields.into(), FieldsShape::Record, attrs)
224            }
225            ast::StructKind::Tuple(it) => {
226                let mut fields = vec![];
227                let mut attrs = vec![];
228
229                for (i, field) in it.fields().enumerate() {
230                    let data = self.lower_tuple_field(i, &field);
231                    fields.push(data);
232                    let attr = RawAttrs::new(self.db, &field, self.span_map());
233                    if !attr.is_empty() {
234                        attrs.push((i, attr))
235                    }
236                }
237                (fields.into(), FieldsShape::Tuple, attrs)
238            }
239            ast::StructKind::Unit => (Box::default(), FieldsShape::Unit, Vec::default()),
240        }
241    }

281    fn lower_enum(&mut self, enum_: &ast::Enum) -> Option<FileItemTreeId<Enum>> {
282        let visibility = self.lower_visibility(enum_);
283        let name = enum_.name()?.as_name();
284        let ast_id = self.source_ast_id_map.ast_id(enum_);
285        let variants = match &enum_.variant_list() {
286            Some(variant_list) => self.lower_variants(variant_list),
287            None => {
288                FileItemTreeId(self.next_variant_idx())..FileItemTreeId(self.next_variant_idx())
289            }
290        };
291        let res = Enum { name, visibility, variants, ast_id };
292        let id = id(self.data().enums.alloc(res));
293        Some(id)
294    }

296    fn lower_variants(&mut self, variants: &ast::VariantList) -> Range<FileItemTreeId<Variant>> {
297        let start = self.next_variant_idx();
298        for variant in variants.variants() {
299            let idx = self.lower_variant(&variant);
300            self.add_attrs(id(idx).into(), RawAttrs::new(self.db, &variant, self.span_map()));
301        }
302        let end = self.next_variant_idx();
303        FileItemTreeId(start)..FileItemTreeId(end)
304    }

306    fn lower_variant(&mut self, variant: &ast::Variant) -> Idx<Variant> {
307        let name = match variant.name() {
308            Some(name) => name.as_name(),
309            None => Name::missing(),
310        };
311        let (fields, kind, attrs) = self.lower_fields(&variant.kind());
312        let ast_id = self.source_ast_id_map.ast_id(variant);
313        let res = Variant { name, fields, shape: kind, ast_id };
314        let id = self.data().variants.alloc(res);
315        for (idx, attr) in attrs {
316            self.add_attrs(
317                AttrOwner::Field(
318                    FieldParent::EnumVariant(FileItemTreeId(id)),
319                    Idx::from_raw(RawIdx::from_u32(idx as u32)),
320                ),
321                attr,
322            );
323        }
324        id

367    fn lower_module(&mut self, module: &ast::Module) -> Option<FileItemTreeId<Mod>> {
368        let name = module.name()?.as_name();
369        let visibility = self.lower_visibility(module);
370        let kind = if module.semicolon_token().is_some() {
371            ModKind::Outline
372        } else {
373            ModKind::Inline {
374                items: module
375                    .item_list()
376                    .map(|list| list.items().flat_map(|item| self.lower_mod_item(&item)).collect())
377                    .unwrap_or_else(|| {
378                        cov_mark::hit!(name_res_works_for_broken_modules);
379                        Box::new([]) as Box<[_]>
380                    }),
381            }
382        };
383        let ast_id = self.source_ast_id_map.ast_id(module);
384        let res = Mod { name, visibility, kind, ast_id };
385        Some(id(self.data().mods.alloc(res)))
386    }

388    fn lower_trait(&mut self, trait_def: &ast::Trait) -> Option<FileItemTreeId<Trait>> {
389        let name = trait_def.name()?.as_name();
390        let visibility = self.lower_visibility(trait_def);
391        let ast_id = self.source_ast_id_map.ast_id(trait_def);
392
393        let items = trait_def
394            .assoc_item_list()
395            .into_iter()
396            .flat_map(|list| list.assoc_items())
397            .filter_map(|item_node| self.lower_assoc_item(&item_node))
398            .collect();
399
400        let def = Trait { name, visibility, items, ast_id };
401        let id = id(self.data().traits.alloc(def));
402        Some(id)

418    fn lower_impl(&mut self, impl_def: &ast::Impl) -> FileItemTreeId<Impl> {
419        let ast_id = self.source_ast_id_map.ast_id(impl_def);
420        // We cannot use `assoc_items()` here as that does not include macro calls.
421        let items = impl_def
422            .assoc_item_list()
423            .into_iter()
424            .flat_map(|it| it.assoc_items())
425            .filter_map(|item| self.lower_assoc_item(&item))
426            .collect();
427        // Note that trait impls don't get implicit `Self` unlike traits, because here they are a
428        // type alias rather than a type parameter, so this is handled by the resolver.
429        let res = Impl { items, ast_id };
430        id(self.data().impls.alloc(res))
431    }

459    fn lower_macro_call(&mut self, m: &ast::MacroCall) -> Option<FileItemTreeId<MacroCall>> {
460        let span_map = self.span_map();
461        let path = m.path()?;
462        let range = path.syntax().text_range();
463        let path = Interned::new(ModPath::from_src(self.db, path, &mut |range| {
464            span_map.span_for_range(range).ctx

467        let expand_to = hir_expand::ExpandTo::from_call_site(m);
468        let res = MacroCall { path, ast_id, expand_to, ctxt: span_map.span_for_range(range).ctx };
469        Some(id(self.data().macro_calls.alloc(res)))

490    fn lower_extern_block(&mut self, block: &ast::ExternBlock) -> FileItemTreeId<ExternBlock> {
491        let ast_id = self.source_ast_id_map.ast_id(block);
492        let abi = block.abi().map(lower_abi);
493        let children: Box<[_]> = block.extern_item_list().map_or(Box::new([]), |list| {
494            list.extern_items()
495                .filter_map(|item| {
496                    // Note: All items in an `extern` block need to be lowered as if they're outside of one
497                    // (in other words, the knowledge that they're in an extern block must not be used).
498                    // This is because an extern block can contain macros whose ItemTree's top-level items
499                    // should be considered to be in an extern block too.
500                    let mod_item: ModItem = match &item {
501                        ast::ExternItem::Fn(ast) => self.lower_function(ast)?.into(),
502                        ast::ExternItem::Static(ast) => self.lower_static(ast)?.into(),
503                        ast::ExternItem::TypeAlias(ty) => self.lower_type_alias(ty)?.into(),
504                        ast::ExternItem::MacroCall(call) => self.lower_macro_call(call)?.into(),
505                    };
506                    let attrs = RawAttrs::new(self.db, &item, self.span_map());
507                    self.add_attrs(mod_item.into(), attrs);
508                    Some(mod_item)
509                })
510                .collect()
511        });
512
513        let res = ExternBlock { abi, ast_id, children };
514        id(self.data().extern_blocks.alloc(res))
515    }

517    fn lower_visibility(&mut self, item: &dyn ast::HasVisibility) -> RawVisibilityId {

519            self.span_map().span_for_range(range).ctx

524    fn next_variant_idx(&self) -> Idx<Variant> {

531fn lower_abi(abi: ast::Abi) -> Symbol {
532    match abi.abi_string() {
533        Some(tok) => Symbol::intern(tok.text_without_quotes()),
534        // `extern` default to be `extern "C"`.
535        _ => sym::C.clone(),
536    }

539struct UseTreeLowering<'a> {
540    db: &'a dyn DefDatabase,
541    mapping: Arena<ast::UseTree>,

544impl UseTreeLowering<'_> {
545    fn lower_use_tree(
546        &mut self,
547        tree: ast::UseTree,
548        span_for_range: &mut dyn FnMut(::tt::TextRange) -> SyntaxContext,
549    ) -> Option<UseTree> {
550        if let Some(use_tree_list) = tree.use_tree_list() {
551            let prefix = match tree.path() {
552                // E.g. use something::{{{inner}}};
553                None => None,
554                // E.g. `use something::{inner}` (prefix is `None`, path is `something`)
555                // or `use something::{path::{inner::{innerer}}}` (prefix is `something::path`, path is `inner`)
556                Some(path) => {
557                    match ModPath::from_src(self.db, path, span_for_range) {
558                        Some(it) => Some(it),
559                        None => return None, // FIXME: report errors somewhere
560                    }
561                }
562            };
563
564            let list = use_tree_list
565                .use_trees()
566                .filter_map(|tree| self.lower_use_tree(tree, span_for_range))
567                .collect();
568
569            Some(
570                self.use_tree(
571                    UseTreeKind::Prefixed { prefix: prefix.map(Interned::new), list },
572                    tree,
573                ),
574            )
575        } else {
576            let is_glob = tree.star_token().is_some();
577            let path = match tree.path() {
578                Some(path) => Some(ModPath::from_src(self.db, path, span_for_range)?),
579                None => None,
580            };
581            let alias = tree.rename().map(|a| {
582                a.name().map(|it| it.as_name()).map_or(ImportAlias::Underscore, ImportAlias::Alias)
583            });
584            if alias.is_some() && is_glob {
585                return None;
586            }
587
588            match (path, alias, is_glob) {
589                (path, None, true) => {
590                    if path.is_none() {
591                        cov_mark::hit!(glob_enum_group);
592                    }
593                    Some(self.use_tree(UseTreeKind::Glob { path: path.map(Interned::new) }, tree))
594                }
595                // Globs can't be renamed
596                (_, Some(_), true) | (None, None, false) => None,
597                // `bla::{ as Name}` is invalid
598                (None, Some(_), false) => None,
599                (Some(path), alias, false) => Some(
600                    self.use_tree(UseTreeKind::Single { path: Interned::new(path), alias }, tree),
601                ),
602            }
603        }
604    }
605
606    fn use_tree(&mut self, kind: UseTreeKind, ast: ast::UseTree) -> UseTree {
607        let index = self.mapping.alloc(ast);
608        UseTree { index, kind }
609    }
610}
611
612pub(crate) fn lower_use_tree(
613    db: &dyn DefDatabase,
614    tree: ast::UseTree,
615    span_for_range: &mut dyn FnMut(::tt::TextRange) -> SyntaxContext,
616) -> Option<(UseTree, Arena<ast::UseTree>)> {
617    let mut lowering = UseTreeLowering { db, mapping: Arena::new() };
618    let tree = lowering.lower_use_tree(tree, span_for_range)?;
619    Some((tree, lowering.mapping))
620}

622fn private_vis() -> RawVisibility {
623    RawVisibility::Module(
624        Interned::new(ModPath::from_kind(PathKind::SELF)),
625        VisibilityExplicitness::Implicit,
626    )
627}
628
629fn visibility_from_ast(
630    db: &dyn DefDatabase,
631    node: Option<ast::Visibility>,
632    span_for_range: &mut dyn FnMut(::tt::TextRange) -> SyntaxContext,
633) -> RawVisibility {
634    let Some(node) = node else { return private_vis() };
635    let path = match node.kind() {
636        ast::VisibilityKind::In(path) => {
637            let path = ModPath::from_src(db, path, span_for_range);
638            match path {
639                None => return private_vis(),
640                Some(path) => path,
641            }
642        }
643        ast::VisibilityKind::PubCrate => ModPath::from_kind(PathKind::Crate),
644        ast::VisibilityKind::PubSuper => ModPath::from_kind(PathKind::Super(1)),
645        ast::VisibilityKind::PubSelf => ModPath::from_kind(PathKind::SELF),
646        ast::VisibilityKind::Pub => return RawVisibility::Public,
647    };
648    RawVisibility::Module(Interned::new(path), VisibilityExplicitness::Explicit)
649}
