1//! Serialization-friendly representation of `tt::TopSubtree`.
2//!
3//! It is possible to serialize `TopSubtree` recursively, as a tree, but using
4//! arbitrary-nested trees in JSON is problematic, as they can cause the JSON
5//! parser to overflow the stack.
6//!
7//! Additionally, such implementation would be pretty verbose, and we do care
8//! about performance here a bit.
9//!
10//! So what this module does is dumping a `tt::TopSubtree` into a bunch of flat
11//! array of numbers. See the test in the parent module to get an example
12//! output.
13//!
14//! ```json
15//!  {
16//!    // Array of subtrees, each subtree is represented by 4 numbers:
17//!    // id of delimiter, delimiter kind, index of first child in `token_tree`,
18//!    // index of last child in `token_tree`
19//!    "subtree":[4294967295,0,0,5,2,2,5,5],
20//!    // 2 ints per literal: [token id, index into `text`]
21//!    "literal":[4294967295,1],
22//!    // 3 ints per punct: [token id, char, spacing]
23//!    "punct":[4294967295,64,1],
24//!    // 2 ints per ident: [token id, index into `text`]
25//!    "ident":   [0,0,1,1],
26//!    // children of all subtrees, concatenated. Each child is represented as `index << 2 | tag`
27//!    // where tag denotes one of subtree, literal, punct or ident.
28//!    "token_tree":[3,7,1,4],
29//!    // Strings shared by idents and literals
30//!    "text": ["struct","Foo"]
31//!  }
32//! ```
33//!
34//! We probably should replace most of the code here with bincode someday, but,
35//! as we don't have bincode in Cargo.toml yet, lets stick with serde_json for
36//! the time being.

47pub type SpanDataIndexMap =
48    indexmap::IndexSet<Span, std::hash::BuildHasherDefault<rustc_hash::FxHasher>>;
49
50pub fn serialize_span_data_index_map(map: &SpanDataIndexMap) -> Vec<u32> {
51    map.iter()
52        .flat_map(|span| {
53            [
54                span.anchor.file_id.as_u32(),
55                span.anchor.ast_id.into_raw(),
56                span.range.start().into(),
57                span.range.end().into(),
58                span.ctx.into_u32(),
59            ]
60        })
61        .collect()
62}
63
64pub fn deserialize_span_data_index_map(map: &[u32]) -> SpanDataIndexMap {
65    debug_assert!(map.len() % 5 == 0);
66    map.chunks_exact(5)
67        .map(|span| {
68            let &[file_id, ast_id, start, end, e] = span else { unreachable!() };
69            Span {
70                anchor: SpanAnchor {
71                    file_id: EditionedFileId::from_raw(file_id),
72                    ast_id: ErasedFileAstId::from_raw(ast_id),
73                },
74                range: TextRange::new(start.into(), end.into()),
75                ctx: SyntaxContext::from_u32(e),
76            }
77        })
78        .collect()
79}

117impl FlatTree {
118    pub fn new(
119        subtree: tt::SubtreeView<'_, Span>,
120        version: u32,
121        span_data_table: &mut SpanDataIndexMap,
122    ) -> FlatTree {
123        let mut w = Writer {
124            string_table: FxHashMap::default(),
125            work: VecDeque::new(),
126            span_data_table,
127
128            subtree: Vec::new(),
129            literal: Vec::new(),
130            punct: Vec::new(),
131            ident: Vec::new(),
132            token_tree: Vec::new(),
133            text: Vec::new(),
134            version,
135        };
136        w.write(subtree);
137
138        FlatTree {
139            subtree: if version >= ENCODE_CLOSE_SPAN_VERSION {
140                write_vec(w.subtree, SubtreeRepr::write_with_close_span)
141            } else {
142                write_vec(w.subtree, SubtreeRepr::write)
143            },
144            literal: if version >= EXTENDED_LEAF_DATA {
145                write_vec(w.literal, LiteralRepr::write_with_kind)
146            } else {
147                write_vec(w.literal, LiteralRepr::write)
148            },
149            punct: write_vec(w.punct, PunctRepr::write),
150            ident: if version >= EXTENDED_LEAF_DATA {
151                write_vec(w.ident, IdentRepr::write_with_rawness)
152            } else {
153                write_vec(w.ident, IdentRepr::write)
154            },
155            token_tree: w.token_tree,
156            text: w.text,
157        }
158    }

255fn read_vec<T, F: Fn([u32; N]) -> T, const N: usize>(xs: Vec<u32>, f: F) -> Vec<T> {
256    let mut chunks = xs.chunks_exact(N);
257    let res = chunks.by_ref().map(|chunk| f(chunk.try_into().unwrap())).collect();
258    assert!(chunks.remainder().is_empty());
259    res
260}
261
262fn write_vec<T, F: Fn(T) -> [u32; N], const N: usize>(xs: Vec<T>, f: F) -> Vec<u32> {
263    xs.into_iter().flat_map(f).collect()
264}

266impl SubtreeRepr {
267    fn write(self) -> [u32; 4] {
268        let kind = match self.kind {
269            tt::DelimiterKind::Invisible => 0,
270            tt::DelimiterKind::Parenthesis => 1,
271            tt::DelimiterKind::Brace => 2,
272            tt::DelimiterKind::Bracket => 3,
273        };
274        [self.open.0, kind, self.tt[0], self.tt[1]]
275    }
276    fn read([open, kind, lo, len]: [u32; 4]) -> SubtreeRepr {
277        let kind = match kind {
278            0 => tt::DelimiterKind::Invisible,
279            1 => tt::DelimiterKind::Parenthesis,
280            2 => tt::DelimiterKind::Brace,
281            3 => tt::DelimiterKind::Bracket,
282            other => panic!("bad kind {other}"),
283        };
284        SubtreeRepr { open: TokenId(open), close: TokenId(!0), kind, tt: [lo, len] }
285    }
286    fn write_with_close_span(self) -> [u32; 5] {
287        let kind = match self.kind {
288            tt::DelimiterKind::Invisible => 0,
289            tt::DelimiterKind::Parenthesis => 1,
290            tt::DelimiterKind::Brace => 2,
291            tt::DelimiterKind::Bracket => 3,
292        };
293        [self.open.0, self.close.0, kind, self.tt[0], self.tt[1]]
294    }
295    fn read_with_close_span([open, close, kind, lo, len]: [u32; 5]) -> SubtreeRepr {
296        let kind = match kind {
297            0 => tt::DelimiterKind::Invisible,
298            1 => tt::DelimiterKind::Parenthesis,
299            2 => tt::DelimiterKind::Brace,
300            3 => tt::DelimiterKind::Bracket,
301            other => panic!("bad kind {other}"),
302        };
303        SubtreeRepr { open: TokenId(open), close: TokenId(close), kind, tt: [lo, len] }
304    }
305}

322impl PunctRepr {
323    fn write(self) -> [u32; 3] {
324        let spacing = match self.spacing {
325            tt::Spacing::Alone | tt::Spacing::JointHidden => 0,
326            tt::Spacing::Joint => 1,
327        };
328        [self.id.0, self.char as u32, spacing]
329    }
330    fn read([id, char, spacing]: [u32; 3]) -> PunctRepr {
331        let spacing = match spacing {
332            0 => tt::Spacing::Alone,
333            1 => tt::Spacing::Joint,
334            other => panic!("bad spacing {other}"),
335        };
336        PunctRepr { id: TokenId(id), char: char.try_into().unwrap(), spacing }
337    }

342        [self.id.0, self.text]

355trait InternableSpan: Copy {
356    type Table;
357    fn token_id_of(table: &mut Self::Table, s: Self) -> TokenId;
358    fn span_for_token_id(table: &Self::Table, id: TokenId) -> Self;
359}
360
361impl InternableSpan for TokenId {
362    type Table = ();
363    fn token_id_of((): &mut Self::Table, token_id: Self) -> TokenId {
364        token_id
365    }
366
367    fn span_for_token_id((): &Self::Table, id: TokenId) -> Self {
368        id
369    }
370}
371impl InternableSpan for Span {
372    type Table = SpanDataIndexMap;
373    fn token_id_of(table: &mut Self::Table, span: Self) -> TokenId {
374        TokenId(table.insert_full(span).0 as u32)
375    }
376    fn span_for_token_id(table: &Self::Table, id: TokenId) -> Self {
377        *table.get_index(id.0 as usize).unwrap_or_else(|| &table[0])
378    }
379}

381struct Writer<'a, 'span, S: InternableSpan> {
382    work: VecDeque<(usize, tt::iter::TtIter<'a, S>)>,
383    string_table: FxHashMap<std::borrow::Cow<'a, str>, u32>,
384    span_data_table: &'span mut S::Table,
385    version: u32,
386
387    subtree: Vec<SubtreeRepr>,
388    literal: Vec<LiteralRepr>,
389    punct: Vec<PunctRepr>,
390    ident: Vec<IdentRepr>,
391    token_tree: Vec<u32>,
392    text: Vec<String>,
393}
394
395impl<'a, S: InternableSpan> Writer<'a, '_, S> {
396    fn write(&mut self, root: tt::SubtreeView<'a, S>) {
397        let subtree = root.top_subtree();
398        self.enqueue(subtree, root.iter());
399        while let Some((idx, subtree)) = self.work.pop_front() {
400            self.subtree(idx, subtree);
401        }
402    }
403
404    fn token_id_of(&mut self, span: S) -> TokenId {
405        S::token_id_of(self.span_data_table, span)
406    }
407
408    fn subtree(&mut self, idx: usize, subtree: tt::iter::TtIter<'a, S>) {
409        let mut first_tt = self.token_tree.len();
410        let n_tt = subtree.clone().count(); // FIXME: `count()` walks over the entire iterator.
411        self.token_tree.resize(first_tt + n_tt, !0);
412
413        self.subtree[idx].tt = [first_tt as u32, (first_tt + n_tt) as u32];
414
415        for child in subtree {
416            let idx_tag = match child {
417                tt::iter::TtElement::Subtree(subtree, subtree_iter) => {
418                    let idx = self.enqueue(subtree, subtree_iter);
419                    idx << 2
420                }
421                tt::iter::TtElement::Leaf(leaf) => match leaf {
422                    tt::Leaf::Literal(lit) => {
423                        let idx = self.literal.len() as u32;
424                        let id = self.token_id_of(lit.span);
425                        let (text, suffix) = if self.version >= EXTENDED_LEAF_DATA {
426                            (
427                                self.intern(lit.symbol.as_str()),
428                                lit.suffix.as_ref().map(|s| self.intern(s.as_str())).unwrap_or(!0),
429                            )
430                        } else {
431                            (self.intern_owned(format!("{lit}")), !0)
432                        };
433                        self.literal.push(LiteralRepr {
434                            id,
435                            text,
436                            kind: u16::from_le_bytes(match lit.kind {
437                                tt::LitKind::Err(_) => [0, 0],
438                                tt::LitKind::Byte => [1, 0],
439                                tt::LitKind::Char => [2, 0],
440                                tt::LitKind::Integer => [3, 0],
441                                tt::LitKind::Float => [4, 0],
442                                tt::LitKind::Str => [5, 0],
443                                tt::LitKind::StrRaw(r) => [6, r],
444                                tt::LitKind::ByteStr => [7, 0],
445                                tt::LitKind::ByteStrRaw(r) => [8, r],
446                                tt::LitKind::CStr => [9, 0],
447                                tt::LitKind::CStrRaw(r) => [10, r],
448                            }),
449                            suffix,
450                        });
451                        (idx << 2) | 0b01
452                    }
453                    tt::Leaf::Punct(punct) => {
454                        let idx = self.punct.len() as u32;
455                        let id = self.token_id_of(punct.span);
456                        self.punct.push(PunctRepr { char: punct.char, spacing: punct.spacing, id });
457                        (idx << 2) | 0b10
458                    }
459                    tt::Leaf::Ident(ident) => {
460                        let idx = self.ident.len() as u32;
461                        let id = self.token_id_of(ident.span);
462                        let text = if self.version >= EXTENDED_LEAF_DATA {
463                            self.intern(ident.sym.as_str())
464                        } else if ident.is_raw.yes() {
465                            self.intern_owned(format!("r#{}", ident.sym.as_str(),))
466                        } else {
467                            self.intern(ident.sym.as_str())
468                        };
469                        self.ident.push(IdentRepr { id, text, is_raw: ident.is_raw.yes() });
470                        (idx << 2) | 0b11
471                    }
472                },
473            };
474            self.token_tree[first_tt] = idx_tag;
475            first_tt += 1;
476        }
477    }
478
479    fn enqueue(&mut self, subtree: &'a tt::Subtree<S>, contents: tt::iter::TtIter<'a, S>) -> u32 {
480        let idx = self.subtree.len();
481        let open = self.token_id_of(subtree.delimiter.open);
482        let close = self.token_id_of(subtree.delimiter.close);
483        let delimiter_kind = subtree.delimiter.kind;
484        self.subtree.push(SubtreeRepr { open, close, kind: delimiter_kind, tt: [!0, !0] });
485        self.work.push_back((idx, contents));
486        idx as u32
487    }
488
489    pub(crate) fn intern(&mut self, text: &'a str) -> u32 {
490        let table = &mut self.text;
491        *self.string_table.entry(text.into()).or_insert_with(|| {
492            let idx = table.len();
493            table.push(text.to_owned());
494            idx as u32
495        })
496    }
497
498    pub(crate) fn intern_owned(&mut self, text: String) -> u32 {
499        let table = &mut self.text;
500        *self.string_table.entry(text.clone().into()).or_insert_with(|| {
501            let idx = table.len();
502            table.push(text);
503            idx as u32
504        })
505    }
506}

508struct Reader<'span, S: InternableSpan> {
509    version: u32,
510    subtree: Vec<SubtreeRepr>,
511    literal: Vec<LiteralRepr>,
512    punct: Vec<PunctRepr>,
513    ident: Vec<IdentRepr>,
514    token_tree: Vec<u32>,
515    text: Vec<String>,
516    span_data_table: &'span S::Table,
517}
518
519impl<S: InternableSpan> Reader<'_, S> {
520    pub(crate) fn read(self) -> tt::TopSubtree<S> {
521        let mut res: Vec<Option<(tt::Delimiter<S>, Vec<tt::TokenTree<S>>)>> =
522            vec![None; self.subtree.len()];
523        let read_span = |id| S::span_for_token_id(self.span_data_table, id);
524        for i in (0..self.subtree.len()).rev() {
525            let repr = &self.subtree[i];
526            let token_trees = &self.token_tree[repr.tt[0] as usize..repr.tt[1] as usize];
527            let delimiter = tt::Delimiter {
528                open: read_span(repr.open),
529                close: read_span(repr.close),
530                kind: repr.kind,
531            };
532            let mut s = Vec::new();
533            for &idx_tag in token_trees {
534                let tag = idx_tag & 0b11;
535                let idx = (idx_tag >> 2) as usize;
536                match tag {
537                    // XXX: we iterate subtrees in reverse to guarantee
538                    // that this unwrap doesn't fire.
539                    0b00 => {
540                        let (delimiter, subtree) = res[idx].take().unwrap();
541                        s.push(tt::TokenTree::Subtree(tt::Subtree {
542                            delimiter,
543                            len: subtree.len() as u32,
544                        }));
545                        s.extend(subtree)
546                    }
547                    0b01 => {
548                        use tt::LitKind::*;
549                        let repr = &self.literal[idx];
550                        let text = self.text[repr.text as usize].as_str();
551                        let span = read_span(repr.id);
552                        s.push(
553                            tt::Leaf::Literal(if self.version >= EXTENDED_LEAF_DATA {
554                                tt::Literal {
555                                    symbol: Symbol::intern(text),
556                                    span,
557                                    kind: match u16::to_le_bytes(repr.kind) {
558                                        [0, _] => Err(()),
559                                        [1, _] => Byte,
560                                        [2, _] => Char,
561                                        [3, _] => Integer,
562                                        [4, _] => Float,
563                                        [5, _] => Str,
564                                        [6, r] => StrRaw(r),
565                                        [7, _] => ByteStr,
566                                        [8, r] => ByteStrRaw(r),
567                                        [9, _] => CStr,
568                                        [10, r] => CStrRaw(r),
569                                        _ => unreachable!(),
570                                    },
571                                    suffix: if repr.suffix != !0 {
572                                        Some(Symbol::intern(
573                                            self.text[repr.suffix as usize].as_str(),
574                                        ))
575                                    } else {
576                                        None
577                                    },
578                                }
579                            } else {
580                                tt::token_to_literal(text, span)
581                            })
582                            .into(),
583                        )
584                    }
585                    0b10 => {
586                        let repr = &self.punct[idx];
587                        s.push(
588                            tt::Leaf::Punct(tt::Punct {
589                                char: repr.char,
590                                spacing: repr.spacing,
591                                span: read_span(repr.id),
592                            })
593                            .into(),
594                        )
595                    }
596                    0b11 => {
597                        let repr = &self.ident[idx];
598                        let text = self.text[repr.text as usize].as_str();
599                        let (is_raw, text) = if self.version >= EXTENDED_LEAF_DATA {
600                            (
601                                if repr.is_raw { tt::IdentIsRaw::Yes } else { tt::IdentIsRaw::No },
602                                text,
603                            )
604                        } else {
605                            tt::IdentIsRaw::split_from_symbol(text)
606                        };
607                        s.push(
608                            tt::Leaf::Ident(tt::Ident {
609                                sym: Symbol::intern(text),
610                                span: read_span(repr.id),
611                                is_raw,
612                            })
613                            .into(),
614                        )
615                    }
616                    other => panic!("bad tag: {other}"),
617                }
618            }
619            res[i] = Some((delimiter, s));
620        }
621
622        let (delimiter, mut res) = res[0].take().unwrap();
623        res.insert(0, tt::TokenTree::Subtree(tt::Subtree { delimiter, len: res.len() as u32 }));
624        tt::TopSubtree(res.into_boxed_slice())
625    }
626}
