1//! Diagnostics rendering and fixits.
2//!
3//! Most of the diagnostics originate from the dark depth of the compiler, and
4//! are originally expressed in term of IR. When we emit the diagnostic, we are
5//! usually not in the position to decide how to best "render" it in terms of
6//! user-authored source code. We are especially not in the position to offer
7//! fixits, as the compiler completely lacks the infrastructure to edit the
8//! source code.
9//!
10//! Instead, we "bubble up" raw, structured diagnostics until the `hir` crate,
11//! where we "cook" them so that each diagnostic is formulated in terms of `hir`
12//! types. Well, at least that's the aspiration, the "cooking" is somewhat
13//! ad-hoc at the moment. Anyways, we get a bunch of ide-friendly diagnostic
14//! structs from hir, and we want to render them to unified serializable
15//! representation (span, level, message) here. If we can, we also provide
16//! fixits. By the way, that's why we want to keep diagnostics structured
17//! internally -- so that we have all the info to make fixes.
18//!
19//! We have one "handler" module per diagnostic code. Such a module contains
20//! rendering, optional fixes and tests. It's OK if some low-level compiler
21//! functionality ends up being tested via a diagnostic.
22//!
23//! There are also a couple of ad-hoc diagnostics implemented directly here, we
24//! don't yet have a great pattern for how to do them properly.
25

106// FIXME: Make this an enum
107#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash)]
108pub enum DiagnosticCode {
109    RustcHardError(&'static str),
110    SyntaxError,
111    RustcLint(&'static str),
112    Clippy(&'static str),
113    Ra(&'static str, Severity),
114}
115
116impl DiagnosticCode {
117    pub fn url(&self) -> String {
118        match self {
119            DiagnosticCode::RustcHardError(e) => {
120                format!("https://doc.rust-lang.org/stable/error_codes/{e}.html")
121            }
122            DiagnosticCode::SyntaxError => {
123                String::from("https://doc.rust-lang.org/stable/reference/")
124            }
125            DiagnosticCode::RustcLint(e) => {
126                format!("https://doc.rust-lang.org/rustc/?search={e}")
127            }
128            DiagnosticCode::Clippy(e) => {
129                format!("https://rust-lang.github.io/rust-clippy/master/#/{e}")
130            }
131            DiagnosticCode::Ra(e, _) => {
132                format!("https://rust-analyzer.github.io/manual.html#{e}")
133            }
134        }
135    }
136
137    pub fn as_str(&self) -> &'static str {
138        match self {
139            DiagnosticCode::RustcHardError(r)
140            | DiagnosticCode::RustcLint(r)
141            | DiagnosticCode::Clippy(r)
142            | DiagnosticCode::Ra(r, _) => r,
143            DiagnosticCode::SyntaxError => "syntax-error",
144        }
145    }
146}

148#[derive(Debug)]
149pub struct Diagnostic {
150    pub code: DiagnosticCode,
151    pub message: String,
152    pub range: FileRange,
153    pub severity: Severity,
154    pub unused: bool,
155    pub experimental: bool,
156    pub fixes: Option<Vec<Assist>>,
157    // The node that will be affected by `#[allow]` and similar attributes.
158    pub main_node: Option<InFile<SyntaxNodePtr>>,
159}
160
161impl Diagnostic {
162    fn new(
163        code: DiagnosticCode,
164        message: impl Into<String>,
165        range: impl Into<FileRange>,
166    ) -> Diagnostic {
167        let message = message.into();
168        Diagnostic {
169            code,
170            message,
171            range: range.into(),
172            severity: match code {
173                DiagnosticCode::RustcHardError(_) | DiagnosticCode::SyntaxError => Severity::Error,
174                // FIXME: Rustc lints are not always warning, but the ones that are currently implemented are all warnings.
175                DiagnosticCode::RustcLint(_) => Severity::Warning,
176                // FIXME: We can make this configurable, and if the user uses `cargo clippy` on flycheck, we can
177                // make it normal warning.
178                DiagnosticCode::Clippy(_) => Severity::WeakWarning,
179                DiagnosticCode::Ra(_, s) => s,
180            },
181            unused: false,
182            experimental: false,
183            fixes: None,
184            main_node: None,
185        }
186    }
187
188    fn new_with_syntax_node_ptr(
189        ctx: &DiagnosticsContext<'_>,
190        code: DiagnosticCode,
191        message: impl Into<String>,
192        node: InFile<SyntaxNodePtr>,
193    ) -> Diagnostic {
194        Diagnostic::new(code, message, ctx.sema.diagnostics_display_range(node))
195            .with_main_node(node)
196    }
197
198    fn experimental(mut self) -> Diagnostic {
199        self.experimental = true;
200        self
201    }
202
203    fn with_main_node(mut self, main_node: InFile<SyntaxNodePtr>) -> Diagnostic {
204        self.main_node = Some(main_node);
205        self
206    }
207
208    fn with_fixes(mut self, fixes: Option<Vec<Assist>>) -> Diagnostic {
209        self.fixes = fixes;
210        self
211    }
212
213    fn with_unused(mut self, unused: bool) -> Diagnostic {
214        self.unused = unused;
215        self
216    }
217}

280struct DiagnosticsContext<'a> {
281    config: &'a DiagnosticsConfig,
282    sema: Semantics<'a, RootDatabase>,
283    resolve: &'a AssistResolveStrategy,
284    edition: Edition,
285    display_target: DisplayTarget,
286    is_nightly: bool,
287}
288
289impl DiagnosticsContext<'_> {
290    fn resolve_precise_location(
291        &self,
292        node: &InFile<SyntaxNodePtr>,
293        precise_location: Option<TextRange>,
294    ) -> FileRange {
295        let sema = &self.sema;
296        (|| {
297            let precise_location = precise_location?;
298            let root = sema.parse_or_expand(node.file_id);
299            match root.covering_element(precise_location) {
300                syntax::NodeOrToken::Node(it) => Some(sema.original_range(&it)),
301                syntax::NodeOrToken::Token(it) => {
302                    node.with_value(it).original_file_range_opt(sema.db)
303                }
304            }
305        })()
306        .unwrap_or_else(|| sema.diagnostics_display_range(*node))
307        .into()
308    }
309}

311/// Request parser level diagnostics for the given [`FileId`].
312pub fn syntax_diagnostics(
313    db: &RootDatabase,
314    config: &DiagnosticsConfig,
315    file_id: FileId,
316) -> Vec<Diagnostic> {
317    let _p = tracing::info_span!("syntax_diagnostics").entered();
318
319    if config.disabled.contains("syntax-error") {
320        return Vec::new();
321    }
322
323    let sema = Semantics::new(db);
324    let editioned_file_id = sema
325        .attach_first_edition(file_id)
326        .unwrap_or_else(|| EditionedFileId::current_edition(file_id));
327
328    let (file_id, _) = editioned_file_id.unpack();
329
330    let editioned_file_id_wrapper =
331        ide_db::base_db::EditionedFileId::new(db.as_dyn_database(), editioned_file_id);
332
333    // [#3434] Only take first 128 errors to prevent slowing down editor/ide, the number 128 is chosen arbitrarily.
334    db.parse_errors(editioned_file_id_wrapper)
335        .into_iter()
336        .flatten()
337        .take(128)
338        .map(|err| {
339            Diagnostic::new(
340                DiagnosticCode::SyntaxError,
341                format!("Syntax Error: {err}"),
342                FileRange { file_id, range: err.range() },
343            )
344        })
345        .collect()
346}

348/// Request semantic diagnostics for the given [`FileId`]. The produced diagnostics may point to other files
349/// due to macros.
350pub fn semantic_diagnostics(
351    db: &RootDatabase,
352    config: &DiagnosticsConfig,
353    resolve: &AssistResolveStrategy,
354    file_id: FileId,
355) -> Vec<Diagnostic> {
356    let _p = tracing::info_span!("semantic_diagnostics").entered();
357    let sema = Semantics::new(db);
358    let editioned_file_id = sema
359        .attach_first_edition(file_id)
360        .unwrap_or_else(|| EditionedFileId::current_edition(file_id));
361
362    let (file_id, edition) = editioned_file_id.unpack();
363    let editioned_file_id_wrapper =
364        ide_db::base_db::EditionedFileId::new(db.as_dyn_database(), editioned_file_id);
365
366    let mut res = Vec::new();
367
368    let parse = sema.parse(editioned_file_id_wrapper);
369
370    // FIXME: This iterates the entire file which is a rather expensive operation.
371    // We should implement these differently in some form?
372    // Salsa caching + incremental re-parse would be better here
373    for node in parse.syntax().descendants() {
374        handlers::useless_braces::useless_braces(&mut res, editioned_file_id, &node);
375        handlers::field_shorthand::field_shorthand(&mut res, editioned_file_id, &node);
376        handlers::json_is_not_rust::json_in_items(
377            &sema,
378            &mut res,
379            editioned_file_id,
380            &node,
381            config,
382            edition,
383        );
384    }
385
386    let module = sema.file_to_module_def(file_id);
387
388    let is_nightly = matches!(
389        module.and_then(|m| db.toolchain_channel(m.krate().into())),
390        Some(ReleaseChannel::Nightly) | None
391    );
392
393    let krate = match module {
394        Some(module) => module.krate(),
395        None => {
396            match db.all_crates().last() {
397                Some(last) => (*last).into(),
398                // short-circuit, return an empty vec of diagnostics
399                None => return vec![],
400            }
401        }
402    };
403    let display_target = krate.to_display_target(db);
404    let ctx = DiagnosticsContext { config, sema, resolve, edition, is_nightly, display_target };
405
406    let mut diags = Vec::new();
407    match module {
408        // A bunch of parse errors in a file indicate some bigger structural parse changes in the
409        // file, so we skip semantic diagnostics so we can show these faster.
410        Some(m) => {
411            if db.parse_errors(editioned_file_id_wrapper).is_none_or(|es| es.len() < 16) {
412                m.diagnostics(db, &mut diags, config.style_lints);
413            }
414        }
415        None => handlers::unlinked_file::unlinked_file(&ctx, &mut res, editioned_file_id.file_id()),
416    }
417
418    for diag in diags {
419        let d = match diag {
420            AnyDiagnostic::AwaitOutsideOfAsync(d) => handlers::await_outside_of_async::await_outside_of_async(&ctx, &d),
421            AnyDiagnostic::CastToUnsized(d) => handlers::invalid_cast::cast_to_unsized(&ctx, &d),
422            AnyDiagnostic::ExpectedFunction(d) => handlers::expected_function::expected_function(&ctx, &d),
423            AnyDiagnostic::InactiveCode(d) => match handlers::inactive_code::inactive_code(&ctx, &d) {
424                Some(it) => it,
425                None => continue,
426            }
427            AnyDiagnostic::IncoherentImpl(d) => handlers::incoherent_impl::incoherent_impl(&ctx, &d),
428            AnyDiagnostic::IncorrectCase(d) => handlers::incorrect_case::incorrect_case(&ctx, &d),
429            AnyDiagnostic::InvalidCast(d) => handlers::invalid_cast::invalid_cast(&ctx, &d),
430            AnyDiagnostic::InvalidDeriveTarget(d) => handlers::invalid_derive_target::invalid_derive_target(&ctx, &d),
431            AnyDiagnostic::MacroDefError(d) => handlers::macro_error::macro_def_error(&ctx, &d),
432            AnyDiagnostic::MacroError(d) => handlers::macro_error::macro_error(&ctx, &d),
433            AnyDiagnostic::MacroExpansionParseError(d) => {
434                // FIXME: Point to the correct error span here, not just the macro-call name
435                res.extend(d.errors.iter().take(16).map(|err| {
436                    {
437                        Diagnostic::new(
438                            DiagnosticCode::SyntaxError,
439                            format!("Syntax Error in Expansion: {err}"),
440                            ctx.resolve_precise_location(&d.node.clone(), d.precise_location),
441                        )
442                    }
443                    .experimental()
444                }));
445                continue;
446            },
447            AnyDiagnostic::MalformedDerive(d) => handlers::malformed_derive::malformed_derive(&ctx, &d),
448            AnyDiagnostic::MismatchedArgCount(d) => handlers::mismatched_arg_count::mismatched_arg_count(&ctx, &d),
449            AnyDiagnostic::MissingFields(d) => handlers::missing_fields::missing_fields(&ctx, &d),
450            AnyDiagnostic::MissingMatchArms(d) => handlers::missing_match_arms::missing_match_arms(&ctx, &d),
451            AnyDiagnostic::MissingUnsafe(d) => handlers::missing_unsafe::missing_unsafe(&ctx, &d),
452            AnyDiagnostic::MovedOutOfRef(d) => handlers::moved_out_of_ref::moved_out_of_ref(&ctx, &d),
453            AnyDiagnostic::NeedMut(d) => match handlers::mutability_errors::need_mut(&ctx, &d) {
454                Some(it) => it,
455                None => continue,
456            },
457            AnyDiagnostic::NonExhaustiveLet(d) => handlers::non_exhaustive_let::non_exhaustive_let(&ctx, &d),
458            AnyDiagnostic::NoSuchField(d) => handlers::no_such_field::no_such_field(&ctx, &d),
459            AnyDiagnostic::PrivateAssocItem(d) => handlers::private_assoc_item::private_assoc_item(&ctx, &d),
460            AnyDiagnostic::PrivateField(d) => handlers::private_field::private_field(&ctx, &d),
461            AnyDiagnostic::ReplaceFilterMapNextWithFindMap(d) => handlers::replace_filter_map_next_with_find_map::replace_filter_map_next_with_find_map(&ctx, &d),
462            AnyDiagnostic::TraitImplIncorrectSafety(d) => handlers::trait_impl_incorrect_safety::trait_impl_incorrect_safety(&ctx, &d),
463            AnyDiagnostic::TraitImplMissingAssocItems(d) => handlers::trait_impl_missing_assoc_item::trait_impl_missing_assoc_item(&ctx, &d),
464            AnyDiagnostic::TraitImplRedundantAssocItems(d) => handlers::trait_impl_redundant_assoc_item::trait_impl_redundant_assoc_item(&ctx, &d),
465            AnyDiagnostic::TraitImplOrphan(d) => handlers::trait_impl_orphan::trait_impl_orphan(&ctx, &d),
466            AnyDiagnostic::TypedHole(d) => handlers::typed_hole::typed_hole(&ctx, &d),
467            AnyDiagnostic::TypeMismatch(d) => handlers::type_mismatch::type_mismatch(&ctx, &d),
468            AnyDiagnostic::UndeclaredLabel(d) => handlers::undeclared_label::undeclared_label(&ctx, &d),
469            AnyDiagnostic::UnimplementedBuiltinMacro(d) => handlers::unimplemented_builtin_macro::unimplemented_builtin_macro(&ctx, &d),
470            AnyDiagnostic::UnreachableLabel(d) => handlers::unreachable_label::unreachable_label(&ctx, &d),
471            AnyDiagnostic::UnresolvedAssocItem(d) => handlers::unresolved_assoc_item::unresolved_assoc_item(&ctx, &d),
472            AnyDiagnostic::UnresolvedExternCrate(d) => handlers::unresolved_extern_crate::unresolved_extern_crate(&ctx, &d),
473            AnyDiagnostic::UnresolvedField(d) => handlers::unresolved_field::unresolved_field(&ctx, &d),
474            AnyDiagnostic::UnresolvedIdent(d) => handlers::unresolved_ident::unresolved_ident(&ctx, &d),
475            AnyDiagnostic::UnresolvedImport(d) => handlers::unresolved_import::unresolved_import(&ctx, &d),
476            AnyDiagnostic::UnresolvedMacroCall(d) => handlers::unresolved_macro_call::unresolved_macro_call(&ctx, &d),
477            AnyDiagnostic::UnresolvedMethodCall(d) => handlers::unresolved_method::unresolved_method(&ctx, &d),
478            AnyDiagnostic::UnresolvedModule(d) => handlers::unresolved_module::unresolved_module(&ctx, &d),
479            AnyDiagnostic::UnusedMut(d) => match handlers::mutability_errors::unused_mut(&ctx, &d) {
480                Some(it) => it,
481                None => continue,
482            },
483            AnyDiagnostic::UnusedVariable(d) => match handlers::unused_variables::unused_variables(&ctx, &d) {
484                Some(it) => it,
485                None => continue,
486            },
487            AnyDiagnostic::BreakOutsideOfLoop(d) => handlers::break_outside_of_loop::break_outside_of_loop(&ctx, &d),
488            AnyDiagnostic::MismatchedTupleStructPatArgCount(d) => handlers::mismatched_arg_count::mismatched_tuple_struct_pat_arg_count(&ctx, &d),
489            AnyDiagnostic::RemoveTrailingReturn(d) => match handlers::remove_trailing_return::remove_trailing_return(&ctx, &d) {
490                Some(it) => it,
491                None => continue,
492            },
493            AnyDiagnostic::RemoveUnnecessaryElse(d) => match handlers::remove_unnecessary_else::remove_unnecessary_else(&ctx, &d) {
494                Some(it) => it,
495                None => continue,
496            },
497            AnyDiagnostic::GenericArgsProhibited(d) => {
498                handlers::generic_args_prohibited::generic_args_prohibited(&ctx, &d)
499            }
500            AnyDiagnostic::ParenthesizedGenericArgsWithoutFnTrait(d) => {
501                handlers::parenthesized_generic_args_without_fn_trait::parenthesized_generic_args_without_fn_trait(&ctx, &d)
502            }
503            AnyDiagnostic::BadRtn(d) => handlers::bad_rtn::bad_rtn(&ctx, &d),
504        };
505        res.push(d)
506    }
507
508    res.retain(|d| {
509        !(ctx.config.disabled.contains(d.code.as_str())
510            || ctx.config.disable_experimental && d.experimental)
511    });
512
513    let mut lints = res
514        .iter_mut()
515        .filter(|it| matches!(it.code, DiagnosticCode::Clippy(_) | DiagnosticCode::RustcLint(_)))
516        .filter_map(|it| {
517            Some((
518                it.main_node.map(|ptr| {
519                    ptr.map(|node| node.to_node(&ctx.sema.parse_or_expand(ptr.file_id)))
520                })?,
521                it,
522            ))
523        })
524        .collect::<Vec<_>>();
525
526    // The edition isn't accurate (each diagnostics may have its own edition due to macros),
527    // but it's okay as it's only being used for error recovery.
528    handle_lints(
529        &ctx.sema,
530        &mut FxHashMap::default(),
531        &mut lints,
532        &mut Vec::new(),
533        editioned_file_id.edition(),
534    );
535
536    res.retain(|d| d.severity != Severity::Allow);
537
538    res.retain_mut(|diag| {
539        if let Some(node) = diag
540            .main_node
541            .map(|ptr| ptr.map(|node| node.to_node(&ctx.sema.parse_or_expand(ptr.file_id))))
542        {
543            handle_diag_from_macros(&ctx.sema, diag, &node)
544        } else {
545            true
546        }
547    });
548
549    res
550}

552/// Request both syntax and semantic diagnostics for the given [`FileId`].
553pub fn full_diagnostics(
554    db: &RootDatabase,
555    config: &DiagnosticsConfig,
556    resolve: &AssistResolveStrategy,
557    file_id: FileId,
558) -> Vec<Diagnostic> {
559    let mut res = syntax_diagnostics(db, config, file_id);
560    let sema = semantic_diagnostics(db, config, resolve, file_id);
561    res.extend(sema);
562    res
563}

565/// Returns whether to keep this diagnostic (or remove it).
566fn handle_diag_from_macros(
567    sema: &Semantics<'_, RootDatabase>,
568    diag: &mut Diagnostic,
569    node: &InFile<SyntaxNode>,
570) -> bool {
571    let Some(macro_file) = node.file_id.macro_file() else { return true };
572    let span_map = sema.db.expansion_span_map(macro_file);
573    let mut spans = span_map.spans_for_range(node.text_range());
574    if spans.any(|span| {
575        span.ctx.outer_expn(sema.db).is_some_and(|expansion| {
576            let macro_call =
577                sema.db.lookup_intern_macro_call(expansion.as_macro_file().macro_call_id);
578            // We don't want to show diagnostics for non-local macros at all, but proc macros authors
579            // seem to rely on being able to emit non-warning-free code, so we don't want to show warnings
580            // for them even when the proc macro comes from the same workspace (in rustc that's not a
581            // problem because it doesn't have the concept of workspaces, and proc macros always reside
582            // in a different crate).
583            !Crate::from(macro_call.def.krate).origin(sema.db).is_local()
584                || !macro_call.def.kind.is_declarative()
585        })
586    }) {
587        // Disable suggestions for external macros, they'll change library code and it's just bad.
588        diag.fixes = None;
589
590        // All Clippy lints report in macros, see https://github.com/rust-lang/rust-clippy/blob/903293b199364/declare_clippy_lint/src/lib.rs#L172.
591        if let DiagnosticCode::RustcLint(lint) = diag.code {
592            if !LINTS_TO_REPORT_IN_EXTERNAL_MACROS.contains(lint) {
593                return false;
594            }
595        };
596    }
597    true
598}

600// `__RA_EVERY_LINT` is a fake lint group to allow every lint in proc macros
601
602struct BuiltLint {
603    lint: &'static Lint,
604    groups: Vec<&'static str>,
605}
606
607static RUSTC_LINTS: LazyLock<FxHashMap<&str, BuiltLint>> =
608    LazyLock::new(|| build_lints_map(DEFAULT_LINTS, DEFAULT_LINT_GROUPS, ""));
609
610static CLIPPY_LINTS: LazyLock<FxHashMap<&str, BuiltLint>> = LazyLock::new(|| {
611    build_lints_map(ide_db::generated::lints::CLIPPY_LINTS, CLIPPY_LINT_GROUPS, "clippy::")
612});
613
614// FIXME: Autogenerate this instead of enumerating by hand.
615static LINTS_TO_REPORT_IN_EXTERNAL_MACROS: LazyLock<FxHashSet<&str>> =
616    LazyLock::new(|| FxHashSet::from_iter([]));
617
618fn build_lints_map(
619    lints: &'static [Lint],
620    lint_group: &'static [LintGroup],
621    prefix: &'static str,
622) -> FxHashMap<&'static str, BuiltLint> {
623    let mut map_with_prefixes: FxHashMap<_, _> = lints
624        .iter()
625        .map(|lint| (lint.label, BuiltLint { lint, groups: vec![lint.label, "__RA_EVERY_LINT"] }))
626        .collect();
627    for g in lint_group {
628        let mut add_children = |label: &'static str| {
629            for child in g.children {
630                map_with_prefixes.get_mut(child).unwrap().groups.push(label);
631            }
632        };
633        add_children(g.lint.label);
634
635        if g.lint.label == "nonstandard_style" {
636            // Also add `bad_style`, which for some reason isn't listed in the groups.
637            add_children("bad_style");
638        }
639    }
640    map_with_prefixes.into_iter().map(|(k, v)| (k.strip_prefix(prefix).unwrap(), v)).collect()
641}

643fn handle_lints(
644    sema: &Semantics<'_, RootDatabase>,
645    cache: &mut FxHashMap<HirFileId, FxHashMap<SmolStr, SeverityAttr>>,
646    diagnostics: &mut [(InFile<SyntaxNode>, &mut Diagnostic)],
647    cache_stack: &mut Vec<HirFileId>,
648    edition: Edition,
649) {
650    for (node, diag) in diagnostics {
651        let lint = match diag.code {
652            DiagnosticCode::RustcLint(lint) => RUSTC_LINTS[lint].lint,
653            DiagnosticCode::Clippy(lint) => CLIPPY_LINTS[lint].lint,
654            _ => panic!("non-lint passed to `handle_lints()`"),
655        };
656        let default_severity = default_lint_severity(lint, edition);
657        if !(default_severity == Severity::Allow && diag.severity == Severity::WeakWarning) {
658            diag.severity = default_severity;
659        }
660
661        let mut diag_severity = fill_lint_attrs(sema, node, cache, cache_stack, diag, edition);
662
663        if let outline_diag_severity @ Some(_) =
664            find_outline_mod_lint_severity(sema, node, diag, edition)
665        {
666            diag_severity = outline_diag_severity;
667        }
668
669        if let Some(diag_severity) = diag_severity {
670            diag.severity = diag_severity;
671        }
672    }
673}
674
675fn default_lint_severity(lint: &Lint, edition: Edition) -> Severity {
676    if lint.deny_since.is_some_and(|e| edition >= e) {
677        Severity::Error
678    } else if lint.warn_since.is_some_and(|e| edition >= e) {
679        Severity::Warning
680    } else {
681        lint.default_severity
682    }
683}
684
685fn find_outline_mod_lint_severity(
686    sema: &Semantics<'_, RootDatabase>,
687    node: &InFile<SyntaxNode>,
688    diag: &Diagnostic,
689    edition: Edition,
690) -> Option<Severity> {
691    let mod_node = node.value.ancestors().find_map(ast::Module::cast)?;
692    if mod_node.item_list().is_some() {
693        // Inline modules will be handled by `fill_lint_attrs()`.
694        return None;
695    }
696
697    let mod_def = sema.to_module_def(&mod_node)?;
698    let module_source_file = sema.module_definition_node(mod_def);
699    let mut result = None;
700    let lint_groups = lint_groups(&diag.code, edition);
701    lint_attrs(
702        sema,
703        ast::AnyHasAttrs::cast(module_source_file.value).expect("SourceFile always has attrs"),
704        edition,
705    )
706    .for_each(|(lint, severity)| {
707        if lint_groups.contains(&lint) {
708            result = Some(severity);
709        }
710    });
711    result
712}

714#[derive(Debug, Clone, Copy)]
715struct SeverityAttr {
716    severity: Severity,
717    /// This field counts how far we are from the main node. Bigger values mean more far.
718    ///
719    /// Note this isn't accurate: there can be gaps between values (created when merging severity maps).
720    /// The important thing is that if an attr is closer to the main node, it will have smaller value.
721    ///
722    /// This is necessary even though we take care to never overwrite a value from deeper nesting
723    /// because of lint groups. For example, in the following code:
724    /// ```
725    /// #[warn(non_snake_case)]
726    /// mod foo {
727    ///     #[allow(nonstandard_style)]
728    ///     mod bar {}
729    /// }
730    /// ```
731    /// We want to not warn on non snake case inside `bar`. If we are traversing this for the first
732    /// time, everything will be fine, because we will set `diag_severity` on the first matching group
733    /// and never overwrite it since then. But if `bar` is cached, the cache will contain both
734    /// `#[warn(non_snake_case)]` and `#[allow(nonstandard_style)]`, and without this field, we have
735    /// no way of differentiating between the two.
736    depth: u32,
737}
738
739fn fill_lint_attrs(
740    sema: &Semantics<'_, RootDatabase>,
741    node: &InFile<SyntaxNode>,
742    cache: &mut FxHashMap<HirFileId, FxHashMap<SmolStr, SeverityAttr>>,
743    cache_stack: &mut Vec<HirFileId>,
744    diag: &Diagnostic,
745    edition: Edition,
746) -> Option<Severity> {
747    let mut collected_lint_attrs = FxHashMap::<SmolStr, SeverityAttr>::default();
748    let mut diag_severity = None;
749
750    let mut ancestors = node.value.ancestors().peekable();
751    let mut depth = 0;
752    loop {
753        let ancestor = ancestors.next().expect("we always return from top-level nodes");
754        depth += 1;
755
756        if ancestors.peek().is_none() {
757            // We don't want to insert too many nodes into cache, but top level nodes (aka. outline modules
758            // or macro expansions) need to touch the database so they seem like a good fit to cache.
759
760            if let Some(cached) = cache.get_mut(&node.file_id) {
761                // This node (and everything above it) is already cached; the attribute is either here or nowhere.
762
763                // Workaround for the borrow checker.
764                let cached = std::mem::take(cached);
765
766                cached.iter().for_each(|(lint, severity)| {
767                    for item in &*cache_stack {
768                        let node_cache_entry = cache
769                            .get_mut(item)
770                            .expect("we always insert cached nodes into the cache map");
771                        let lint_cache_entry = node_cache_entry.entry(lint.clone());
772                        if let hash_map::Entry::Vacant(lint_cache_entry) = lint_cache_entry {
773                            // Do not overwrite existing lint attributes, as we go bottom to top and bottom attrs
774                            // overwrite top attrs.
775                            lint_cache_entry.insert(SeverityAttr {
776                                severity: severity.severity,
777                                depth: severity.depth + depth,
778                            });
779                        }
780                    }
781                });
782
783                let lints = lint_groups(&diag.code, edition);
784                let all_matching_groups =
785                    lints.iter().filter_map(|lint_group| cached.get(lint_group));
786                let cached_severity =
787                    all_matching_groups.min_by_key(|it| it.depth).map(|it| it.severity);
788
789                cache.insert(node.file_id, cached);
790
791                return diag_severity.or(cached_severity);
792            }
793
794            // Insert this node's descendants' attributes into any outline descendant, but not including this node.
795            // This must come before inserting this node's own attributes to preserve order.
796            collected_lint_attrs.drain().for_each(|(lint, severity)| {
797                if diag_severity.is_none() && lint_groups(&diag.code, edition).contains(&lint) {
798                    diag_severity = Some(severity.severity);
799                }
800
801                for item in &*cache_stack {
802                    let node_cache_entry = cache
803                        .get_mut(item)
804                        .expect("we always insert cached nodes into the cache map");
805                    let lint_cache_entry = node_cache_entry.entry(lint.clone());
806                    if let hash_map::Entry::Vacant(lint_cache_entry) = lint_cache_entry {
807                        // Do not overwrite existing lint attributes, as we go bottom to top and bottom attrs
808                        // overwrite top attrs.
809                        lint_cache_entry.insert(severity);
810                    }
811                }
812            });
813
814            cache_stack.push(node.file_id);
815            cache.insert(node.file_id, FxHashMap::default());
816
817            if let Some(ancestor) = ast::AnyHasAttrs::cast(ancestor) {
818                // Insert this node's attributes into any outline descendant, including this node.
819                lint_attrs(sema, ancestor, edition).for_each(|(lint, severity)| {
820                    if diag_severity.is_none() && lint_groups(&diag.code, edition).contains(&lint) {
821                        diag_severity = Some(severity);
822                    }
823
824                    for item in &*cache_stack {
825                        let node_cache_entry = cache
826                            .get_mut(item)
827                            .expect("we always insert cached nodes into the cache map");
828                        let lint_cache_entry = node_cache_entry.entry(lint.clone());
829                        if let hash_map::Entry::Vacant(lint_cache_entry) = lint_cache_entry {
830                            // Do not overwrite existing lint attributes, as we go bottom to top and bottom attrs
831                            // overwrite top attrs.
832                            lint_cache_entry.insert(SeverityAttr { severity, depth });
833                        }
834                    }
835                });
836            }
837
838            let parent_node = sema.find_parent_file(node.file_id);
839            if let Some(parent_node) = parent_node {
840                let parent_severity =
841                    fill_lint_attrs(sema, &parent_node, cache, cache_stack, diag, edition);
842                if diag_severity.is_none() {
843                    diag_severity = parent_severity;
844                }
845            }
846            cache_stack.pop();
847            return diag_severity;
848        } else if let Some(ancestor) = ast::AnyHasAttrs::cast(ancestor) {
849            lint_attrs(sema, ancestor, edition).for_each(|(lint, severity)| {
850                if diag_severity.is_none() && lint_groups(&diag.code, edition).contains(&lint) {
851                    diag_severity = Some(severity);
852                }
853
854                let lint_cache_entry = collected_lint_attrs.entry(lint);
855                if let hash_map::Entry::Vacant(lint_cache_entry) = lint_cache_entry {
856                    // Do not overwrite existing lint attributes, as we go bottom to top and bottom attrs
857                    // overwrite top attrs.
858                    lint_cache_entry.insert(SeverityAttr { severity, depth });
859                }
860            });
861        }
862    }
863}
864
865fn lint_attrs<'a>(
866    sema: &'a Semantics<'a, RootDatabase>,
867    ancestor: ast::AnyHasAttrs,
868    edition: Edition,
869) -> impl Iterator<Item = (SmolStr, Severity)> + 'a {
870    ancestor
871        .attrs_including_inner()
872        .filter_map(|attr| {
873            attr.as_simple_call().and_then(|(name, value)| match &*name {
874                "allow" | "expect" => Some(Either::Left(iter::once((Severity::Allow, value)))),
875                "warn" => Some(Either::Left(iter::once((Severity::Warning, value)))),
876                "forbid" | "deny" => Some(Either::Left(iter::once((Severity::Error, value)))),
877                "cfg_attr" => {
878                    let mut lint_attrs = Vec::new();
879                    cfg_attr_lint_attrs(sema, &value, &mut lint_attrs);
880                    Some(Either::Right(lint_attrs.into_iter()))
881                }
882                _ => None,
883            })
884        })
885        .flatten()
886        .flat_map(move |(severity, lints)| {
887            parse_tt_as_comma_sep_paths(lints, edition).into_iter().flat_map(move |lints| {
888                // Rejoin the idents with `::`, so we have no spaces in between.
889                lints.into_iter().map(move |lint| {
890                    (
891                        lint.segments().filter_map(|segment| segment.name_ref()).join("::").into(),
892                        severity,
893                    )
894                })
895            })
896        })
897}
898
899fn cfg_attr_lint_attrs(
900    sema: &Semantics<'_, RootDatabase>,
901    value: &ast::TokenTree,
902    lint_attrs: &mut Vec<(Severity, ast::TokenTree)>,
903) {
904    let prev_len = lint_attrs.len();
905
906    let mut iter = value.token_trees_and_tokens().filter(|it| match it {
907        NodeOrToken::Node(_) => true,
908        NodeOrToken::Token(it) => !it.kind().is_trivia(),
909    });
910
911    // Skip the condition.
912    for value in &mut iter {
913        if value.as_token().is_some_and(|it| it.kind() == T![,]) {
914            break;
915        }
916    }
917
918    while let Some(value) = iter.next() {
919        if let Some(token) = value.as_token() {
920            if token.kind() == SyntaxKind::IDENT {
921                let severity = match token.text() {
922                    "allow" | "expect" => Some(Severity::Allow),
923                    "warn" => Some(Severity::Warning),
924                    "forbid" | "deny" => Some(Severity::Error),
925                    "cfg_attr" => {
926                        if let Some(NodeOrToken::Node(value)) = iter.next() {
927                            cfg_attr_lint_attrs(sema, &value, lint_attrs);
928                        }
929                        None
930                    }
931                    _ => None,
932                };
933                if let Some(severity) = severity {
934                    let lints = iter.next();
935                    if let Some(NodeOrToken::Node(lints)) = lints {
936                        lint_attrs.push((severity, lints));
937                    }
938                }
939            }
940        }
941    }
942
943    if prev_len != lint_attrs.len() {
944        if let Some(false) | None = sema.check_cfg_attr(value) {
945            // Discard the attributes when the condition is false.
946            lint_attrs.truncate(prev_len);
947        }
948    }
949}

951#[derive(Debug)]
952struct LintGroups {
953    groups: &'static [&'static str],
954    inside_warnings: bool,
955}
956
957impl LintGroups {
958    fn contains(&self, group: &str) -> bool {
959        self.groups.contains(&group) || (self.inside_warnings && group == "warnings")
960    }
961
962    fn iter(&self) -> impl Iterator<Item = &'static str> {
963        self.groups.iter().copied().chain(self.inside_warnings.then_some("warnings"))
964    }
965}
966
967fn lint_groups(lint: &DiagnosticCode, edition: Edition) -> LintGroups {
968    let (groups, inside_warnings) = match lint {
969        DiagnosticCode::RustcLint(name) => {
970            let lint = &RUSTC_LINTS[name];
971            let inside_warnings = default_lint_severity(lint.lint, edition) == Severity::Warning;
972            (&lint.groups, inside_warnings)
973        }
974        DiagnosticCode::Clippy(name) => {
975            let lint = &CLIPPY_LINTS[name];
976            let inside_warnings = default_lint_severity(lint.lint, edition) == Severity::Warning;
977            (&lint.groups, inside_warnings)
978        }
979        _ => panic!("non-lint passed to `handle_lints()`"),
980    };
981    LintGroups { groups, inside_warnings }
982}

984fn fix(id: &'static str, label: &str, source_change: SourceChange, target: TextRange) -> Assist {
985    let mut res = unresolved_fix(id, label, target);
986    res.source_change = Some(source_change);
987    res
988}
989
990fn unresolved_fix(id: &'static str, label: &str, target: TextRange) -> Assist {
991    assert!(!id.contains(' '));
992    Assist {
993        id: AssistId::quick_fix(id),
994        label: Label::new(label.to_owned()),
995        group: None,
996        target,
997        source_change: None,
998        command: None,
999    }
1000}
