297    pub fn block_expr(
298        &self,
299        statements: impl IntoIterator<Item = ast::Stmt>,
300        tail_expr: Option<ast::Expr>,
301    ) -> ast::BlockExpr {
302        let (statements, mut input) = iterator_input(statements);
303
304        let ast = make::block_expr(statements, tail_expr.clone()).clone_for_update();
305
306        if let Some(mut mapping) = self.mappings() {
307            let stmt_list = ast.stmt_list().unwrap();
308            let mut builder = SyntaxMappingBuilder::new(stmt_list.syntax().clone());
309
310            if let Some(input) = tail_expr {
311                builder.map_node(
312                    input.syntax().clone(),
313                    stmt_list.tail_expr().unwrap().syntax().clone(),
314                );
315            } else if let Some(ast_tail) = stmt_list.tail_expr() {
316                // The parser interpreted the last statement (probably a statement with a block) as an Expr
317                let last_stmt = input.pop().unwrap();
318
319                builder.map_node(last_stmt, ast_tail.syntax().clone());
320            }
321
322            builder.map_children(input, stmt_list.statements().map(|it| it.syntax().clone()));
323
324            builder.finish(&mut mapping);
325        }
326
327        ast
328    }

1149// We need to collect `input` here instead of taking `impl IntoIterator + Clone`,
1150// because if we took `impl IntoIterator + Clone`, that could be something like an
1151// `Iterator::map` with a closure that also makes use of a `SyntaxFactory` constructor.
1152//
1153// In that case, the iterator would be evaluated inside of the call to `map_children`,
1154// and the inner constructor would try to take a mutable borrow of the mappings `RefCell`,
1155// which would panic since it's already being mutably borrowed in the outer constructor.
1156fn iterator_input<N: AstNode>(input: impl IntoIterator<Item = N>) -> (Vec<N>, Vec<SyntaxNode>) {
1157    input
1158        .into_iter()
1159        .map(|it| {
1160            let syntax = it.syntax().clone();
1161            (it, syntax)
1162        })
1163        .collect()
