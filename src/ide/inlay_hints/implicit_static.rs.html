1//! Implementation of "implicit static" inlay hints:
2//! ```no_run
3//! static S: &/* 'static */str = "";
4//! ```
5use either::Either;
6use ide_db::famous_defs::FamousDefs;
7use ide_db::text_edit::TextEdit;
8use span::EditionedFileId;
9use syntax::{
10    SyntaxKind,
11    ast::{self, AstNode},
12};
13
14use crate::{InlayHint, InlayHintPosition, InlayHintsConfig, InlayKind, LifetimeElisionHints};
15
16pub(super) fn hints(
17    acc: &mut Vec<InlayHint>,
18    FamousDefs(_sema, _): &FamousDefs<'_, '_>,
19    config: &InlayHintsConfig,
20    _file_id: EditionedFileId,
21    statik_or_const: Either<ast::Static, ast::Const>,
22) -> Option<()> {
23    if config.lifetime_elision_hints != LifetimeElisionHints::Always {
24        return None;
25    }
26
27    if let Either::Right(it) = &statik_or_const {
28        if ast::AssocItemList::can_cast(
29            it.syntax().parent().map_or(SyntaxKind::EOF, |it| it.kind()),
30        ) {
31            return None;
32        }
33    }
34
35    if let Some(ast::Type::RefType(ty)) = statik_or_const.either(|it| it.ty(), |it| it.ty()) {
36        if ty.lifetime().is_none() {
37            let t = ty.amp_token()?;
38            acc.push(InlayHint {
39                range: t.text_range(),
40                kind: InlayKind::Lifetime,
41                label: "'static".into(),
42                text_edit: Some(config.lazy_text_edit(|| {
43                    TextEdit::insert(t.text_range().start(), "'static ".into())
44                })),
45                position: InlayHintPosition::After,
46                pad_left: false,
47                pad_right: true,
48                resolve_parent: None,
49            });
50        }
51    }
52
53    Some(())
54}
