43#[derive(Debug, Clone, Eq, PartialEq)]
44pub enum TokenExpander {
45    /// Old-style `macro_rules` or the new macros 2.0
46    DeclarativeMacro(Arc<DeclarativeMacroExpander>),
47    /// Stuff like `line!` and `file!`.
48    BuiltIn(BuiltinFnLikeExpander),
49    /// Built-in eagerly expanded fn-like macros (`include!`, `concat!`, etc.)
50    BuiltInEager(EagerExpander),
51    /// `global_allocator` and such.
52    BuiltInAttr(BuiltinAttrExpander),
53    /// `derive(Copy)` and such.
54    BuiltInDerive(BuiltinDeriveExpander),
55    /// The thing we love the most here in rust-analyzer -- procedural macros.
56    ProcMacro(CustomProcMacroExpander),
57}

59#[query_group::query_group]
60pub trait ExpandDatabase: RootQueryDb {
61    /// The proc macros. Do not use this! Use `proc_macros_for_crate()` instead.
62    #[salsa::input]
63    fn proc_macros(&self) -> Arc<ProcMacros>;
64
65    /// Incrementality query to prevent queries from directly depending on `ExpandDatabase::proc_macros`.
66    #[salsa::invoke_actual(crate::proc_macro::proc_macros_for_crate)]
67    fn proc_macros_for_crate(&self, krate: Crate) -> Option<Arc<CrateProcMacros>>;
68
69    fn ast_id_map(&self, file_id: HirFileId) -> Arc<AstIdMap>;
70
71    #[salsa::transparent]
72    fn parse_or_expand(&self, file_id: HirFileId) -> SyntaxNode;
73
74    /// Implementation for the macro case.
75    #[salsa::lru(512)]
76    fn parse_macro_expansion(
77        &self,
78        macro_file: span::MacroFileId,
79    ) -> ExpandResult<(Parse<SyntaxNode>, Arc<ExpansionSpanMap>)>;
80
81    #[salsa::transparent]
82    #[salsa::invoke(SpanMap::new)]
83    fn span_map(&self, file_id: HirFileId) -> SpanMap;
84
85    #[salsa::transparent]
86    #[salsa::invoke(crate::span_map::expansion_span_map)]
87    fn expansion_span_map(&self, file_id: MacroFileId) -> Arc<ExpansionSpanMap>;
88    #[salsa::invoke(crate::span_map::real_span_map)]
89    fn real_span_map(&self, file_id: EditionedFileId) -> Arc<RealSpanMap>;
90
91    /// Macro ids. That's probably the tricksiest bit in rust-analyzer, and the
92    /// reason why we use salsa at all.
93    ///
94    /// We encode macro definitions into ids of macro calls, this what allows us
95    /// to be incremental.
96    #[salsa::transparent]
97    fn intern_macro_call(&self, macro_call: MacroCallLoc) -> MacroCallId;
98    #[salsa::transparent]
99    fn lookup_intern_macro_call(&self, macro_call: MacroCallId) -> MacroCallLoc;
100
101    /// Lowers syntactic macro call to a token tree representation. That's a firewall
102    /// query, only typing in the macro call itself changes the returned
103    /// subtree.
104    #[deprecated = "calling this is incorrect, call `macro_arg_considering_derives` instead"]
105    #[salsa::invoke(macro_arg)]
106    fn macro_arg(&self, id: MacroCallId) -> MacroArgResult;
107
108    #[salsa::transparent]
109    fn macro_arg_considering_derives(
110        &self,
111        id: MacroCallId,
112        kind: &MacroCallKind,
113    ) -> MacroArgResult;
114
115    /// Fetches the expander for this macro.
116    #[salsa::transparent]
117    #[salsa::invoke(TokenExpander::macro_expander)]
118    fn macro_expander(&self, id: MacroDefId) -> TokenExpander;
119
120    /// Fetches (and compiles) the expander of this decl macro.
121    #[salsa::invoke(DeclarativeMacroExpander::expander)]
122    fn decl_macro_expander(
123        &self,
124        def_crate: Crate,
125        id: AstId<ast::Macro>,
126    ) -> Arc<DeclarativeMacroExpander>;
127
128    /// Special case of the previous query for procedural macros. We can't LRU
129    /// proc macros, since they are not deterministic in general, and
130    /// non-determinism breaks salsa in a very, very, very bad way.
131    /// @edwin0cheng heroically debugged this once! See #4315 for details
132    #[salsa::invoke(expand_proc_macro)]
133    fn expand_proc_macro(&self, call: MacroCallId) -> ExpandResult<Arc<tt::TopSubtree>>;
134    /// Retrieves the span to be used for a proc-macro expansions spans.
135    /// This is a firewall query as it requires parsing the file, which we don't want proc-macros to
136    /// directly depend on as that would cause to frequent invalidations, mainly because of the
137    /// parse queries being LRU cached. If they weren't the invalidations would only happen if the
138    /// user wrote in the file that defines the proc-macro.
139    fn proc_macro_span(&self, fun: AstId<ast::Fn>) -> Span;
140
141    /// Firewall query that returns the errors from the `parse_macro_expansion` query.
142    #[salsa::invoke(parse_macro_expansion_error)]
143    fn parse_macro_expansion_error(
144        &self,
145        macro_call: MacroCallId,
146    ) -> Option<Arc<ExpandResult<Arc<[SyntaxError]>>>>;
147
148    #[salsa::transparent]
149    fn syntax_context(&self, file: HirFileId, edition: Edition) -> SyntaxContext;
150}

170fn syntax_context(db: &dyn ExpandDatabase, file: HirFileId, edition: Edition) -> SyntaxContext {
171    match file.repr() {
172        HirFileIdRepr::FileId(_) => SyntaxContext::root(edition),
173        HirFileIdRepr::MacroFile(m) => {
174            let kind = db.lookup_intern_macro_call(m.macro_call_id).kind;
175            db.macro_arg_considering_derives(m.macro_call_id, &kind).2.ctx
176        }
177    }

180/// This expands the given macro call, but with different arguments. This is
181/// used for completion, where we want to see what 'would happen' if we insert a
182/// token. The `token_to_map` mapped down into the expansion, with the mapped
183/// token(s) returned with their priority.
184pub fn expand_speculative(
185    db: &dyn ExpandDatabase,
186    actual_macro_call: MacroCallId,
187    speculative_args: &SyntaxNode,
188    token_to_map: SyntaxToken,
189) -> Option<(SyntaxNode, Vec<(SyntaxToken, u8)>)> {
190    let loc = db.lookup_intern_macro_call(actual_macro_call);
191    let (_, _, span) = db.macro_arg_considering_derives(actual_macro_call, &loc.kind);
192
193    let span_map = RealSpanMap::absolute(span.anchor.file_id);
194    let span_map = SpanMapRef::RealSpanMap(&span_map);
195
196    // Build the subtree and token mapping for the speculative args
197    let (mut tt, undo_info) = match loc.kind {
198        MacroCallKind::FnLike { .. } => (
199            syntax_bridge::syntax_node_to_token_tree(
200                speculative_args,
201                span_map,
202                span,
203                if loc.def.is_proc_macro() {
204                    DocCommentDesugarMode::ProcMacro
205                } else {
206                    DocCommentDesugarMode::Mbe
207                },
208            ),
209            SyntaxFixupUndoInfo::NONE,
210        ),
211        MacroCallKind::Attr { .. } if loc.def.is_attribute_derive() => (
212            syntax_bridge::syntax_node_to_token_tree(
213                speculative_args,
214                span_map,
215                span,
216                DocCommentDesugarMode::ProcMacro,
217            ),
218            SyntaxFixupUndoInfo::NONE,
219        ),
220        MacroCallKind::Derive { derive_attr_index: index, .. }
221        | MacroCallKind::Attr { invoc_attr_index: index, .. } => {
222            let censor = if let MacroCallKind::Derive { .. } = loc.kind {
223                censor_derive_input(index, &ast::Adt::cast(speculative_args.clone())?)
224            } else {
225                attr_source(index, &ast::Item::cast(speculative_args.clone())?)
226                    .into_iter()
227                    .map(|it| it.syntax().clone().into())
228                    .collect()
229            };
230
231            let censor_cfg =
232                cfg_process::process_cfg_attrs(db, speculative_args, &loc).unwrap_or_default();
233            let mut fixups = fixup::fixup_syntax(
234                span_map,
235                speculative_args,
236                span,
237                DocCommentDesugarMode::ProcMacro,
238            );
239            fixups.append.retain(|it, _| match it {
240                syntax::NodeOrToken::Token(_) => true,
241                it => !censor.contains(it) && !censor_cfg.contains(it),
242            });
243            fixups.remove.extend(censor);
244            fixups.remove.extend(censor_cfg);
245
246            (
247                syntax_bridge::syntax_node_to_token_tree_modified(
248                    speculative_args,
249                    span_map,
250                    fixups.append,
251                    fixups.remove,
252                    span,
253                    DocCommentDesugarMode::ProcMacro,
254                ),
255                fixups.undo_info,
256            )
257        }
258    };
259
260    let attr_arg = match loc.kind {
261        MacroCallKind::Attr { invoc_attr_index, .. } => {
262            let attr = if loc.def.is_attribute_derive() {
263                // for pseudo-derive expansion we actually pass the attribute itself only
264                ast::Attr::cast(speculative_args.clone())
265            } else {
266                // Attributes may have an input token tree, build the subtree and map for this as well
267                // then try finding a token id for our token if it is inside this input subtree.
268                let item = ast::Item::cast(speculative_args.clone())?;
269                collect_attrs(&item)
270                    .nth(invoc_attr_index.ast_index())
271                    .and_then(|x| Either::left(x.1))
272            }?;
273            match attr.token_tree() {
274                Some(token_tree) => {
275                    let mut tree = syntax_node_to_token_tree(
276                        token_tree.syntax(),
277                        span_map,
278                        span,
279                        DocCommentDesugarMode::ProcMacro,
280                    );
281                    *tree.top_subtree_delimiter_mut() = tt::Delimiter::invisible_spanned(span);
282
283                    Some(tree)
284                }
285                _ => None,
286            }
287        }
288        _ => None,
289    };
290
291    // Do the actual expansion, we need to directly expand the proc macro due to the attribute args
292    // Otherwise the expand query will fetch the non speculative attribute args and pass those instead.
293    let mut speculative_expansion = match loc.def.kind {
294        MacroDefKind::ProcMacro(ast, expander, _) => {
295            let span = db.proc_macro_span(ast);
296            *tt.top_subtree_delimiter_mut() = tt::Delimiter::invisible_spanned(span);
297            expander.expand(
298                db,
299                loc.def.krate,
300                loc.krate,
301                &tt,
302                attr_arg.as_ref(),
303                span_with_def_site_ctxt(db, span, actual_macro_call, loc.def.edition),
304                span_with_call_site_ctxt(db, span, actual_macro_call, loc.def.edition),
305                span_with_mixed_site_ctxt(db, span, actual_macro_call, loc.def.edition),
306            )
307        }
308        MacroDefKind::BuiltInAttr(_, it) if it.is_derive() => {
309            pseudo_derive_attr_expansion(&tt, attr_arg.as_ref()?, span)
310        }
311        MacroDefKind::Declarative(it) => {
312            db.decl_macro_expander(loc.krate, it).expand_unhygienic(tt, span, loc.def.edition)
313        }
314        MacroDefKind::BuiltIn(_, it) => {
315            it.expand(db, actual_macro_call, &tt, span).map_err(Into::into)
316        }
317        MacroDefKind::BuiltInDerive(_, it) => {
318            it.expand(db, actual_macro_call, &tt, span).map_err(Into::into)
319        }
320        MacroDefKind::BuiltInEager(_, it) => {
321            it.expand(db, actual_macro_call, &tt, span).map_err(Into::into)
322        }
323        MacroDefKind::BuiltInAttr(_, it) => it.expand(db, actual_macro_call, &tt, span),
324    };
325
326    let expand_to = loc.expand_to();
327
328    fixup::reverse_fixups(&mut speculative_expansion.value, &undo_info);
329    let (node, rev_tmap) =
330        token_tree_to_syntax_node(db, &speculative_expansion.value, expand_to, loc.def.edition);
331
332    let syntax_node = node.syntax_node();
333    let token = rev_tmap
334        .ranges_with_span(span_map.span_for_range(token_to_map.text_range()))
335        .filter_map(|(range, ctx)| syntax_node.covering_element(range).into_token().zip(Some(ctx)))
336        .map(|(t, ctx)| {
337            // prefer tokens of the same kind and text, as well as non opaque marked ones
338            // Note the inversion of the score here, as we want to prefer the first token in case
339            // of all tokens having the same score
340            let ranking = ctx.is_opaque(db) as u8
341                + 2 * (t.kind() != token_to_map.kind()) as u8
342                + 4 * ((t.text() != token_to_map.text()) as u8);
343            (t, ranking)
344        })
345        .collect();
346    Some((node.syntax_node(), token))
347}

353/// Main public API -- parses a hir file, not caring whether it's a real
354/// file or a macro expansion.
355fn parse_or_expand(db: &dyn ExpandDatabase, file_id: HirFileId) -> SyntaxNode {
356    match file_id.repr() {
357        HirFileIdRepr::FileId(editioned_file_id) => {
358            let file_id = base_db::EditionedFileId::new(db, editioned_file_id);
359            db.parse(file_id).syntax_node()
360        }
361
362        HirFileIdRepr::MacroFile(macro_file) => {
363            db.parse_macro_expansion(macro_file).value.0.syntax_node()
364        }
365    }
366}

368// FIXME: We should verify that the parsed node is one of the many macro node variants we expect
369// instead of having it be untyped
370fn parse_macro_expansion(
371    db: &dyn ExpandDatabase,
372    macro_file: MacroFileId,
373) -> ExpandResult<(Parse<SyntaxNode>, Arc<ExpansionSpanMap>)> {
374    let _p = tracing::info_span!("parse_macro_expansion").entered();
375    let loc = db.lookup_intern_macro_call(macro_file.macro_call_id);
376    let def_edition = loc.def.edition;
377    let expand_to = loc.expand_to();
378    let mbe::ValueResult { value: (tt, matched_arm), err } =
379        macro_expand(db, macro_file.macro_call_id, loc);
380
381    let (parse, mut rev_token_map) = token_tree_to_syntax_node(
382        db,
383        match &tt {
384            CowArc::Arc(it) => it,
385            CowArc::Owned(it) => it,
386        },
387        expand_to,
388        def_edition,
389    );
390    rev_token_map.matched_arm = matched_arm;
391
392    ExpandResult { value: (parse, Arc::new(rev_token_map)), err }
393}

395fn parse_macro_expansion_error(
396    db: &dyn ExpandDatabase,
397    macro_call_id: MacroCallId,
398) -> Option<Arc<ExpandResult<Arc<[SyntaxError]>>>> {
399    let e: ExpandResult<Arc<[SyntaxError]>> =
400        db.parse_macro_expansion(MacroFileId { macro_call_id }).map(|it| Arc::from(it.0.errors()));
401    if e.value.is_empty() && e.err.is_none() { None } else { Some(Arc::new(e)) }
402}

404pub(crate) fn parse_with_map(
405    db: &dyn ExpandDatabase,
406    file_id: HirFileId,
407) -> (Parse<SyntaxNode>, SpanMap) {
408    match file_id.repr() {
409        HirFileIdRepr::FileId(editioned_file_id) => {
410            let file_id = base_db::EditionedFileId::new(db, editioned_file_id);
411
412            (
413                db.parse(file_id).to_syntax(),
414                SpanMap::RealSpanMap(db.real_span_map(editioned_file_id)),
415            )
416        }
417        HirFileIdRepr::MacroFile(macro_file) => {
418            let (parse, map) = db.parse_macro_expansion(macro_file).value;
419            (parse, SpanMap::ExpansionSpanMap(map))
420        }
421    }
422}

424/// This resolves the [MacroCallId] to check if it is a derive macro if so get the [macro_arg] for the derive.
425/// Other wise return the [macro_arg] for the macro_call_id.
426///
427/// This is not connected to the database so it does not cached the result. However, the inner [macro_arg] query is
428#[allow(deprecated)] // we are macro_arg_considering_derives
429fn macro_arg_considering_derives(
430    db: &dyn ExpandDatabase,
431    id: MacroCallId,
432    kind: &MacroCallKind,
433) -> MacroArgResult {
434    match kind {
435        // Get the macro arg for the derive macro
436        MacroCallKind::Derive { derive_macro_id, .. } => db.macro_arg(*derive_macro_id),
437        // Normal macro arg
438        _ => db.macro_arg(id),
439    }
440}

442fn macro_arg(db: &dyn ExpandDatabase, id: MacroCallId) -> MacroArgResult {
443    let loc = db.lookup_intern_macro_call(id);
444
445    if let MacroCallLoc {
446        def: MacroDefId { kind: MacroDefKind::BuiltInEager(..), .. },
447        kind: MacroCallKind::FnLike { eager: Some(eager), .. },
448        ..
449    } = &loc
450    {
451        return (eager.arg.clone(), SyntaxFixupUndoInfo::NONE, eager.span);
452    }
453
454    let (parse, map) = parse_with_map(db, loc.kind.file_id());
455    let root = parse.syntax_node();
456
457    let (censor, item_node, span) = match loc.kind {
458        MacroCallKind::FnLike { ast_id, .. } => {
459            let node = &ast_id.to_ptr(db).to_node(&root);
460            let path_range = node
461                .path()
462                .map_or_else(|| node.syntax().text_range(), |path| path.syntax().text_range());
463            let span = map.span_for_range(path_range);
464
465            let dummy_tt = |kind| {
466                (
467                    Arc::new(tt::TopSubtree::from_token_trees(
468                        tt::Delimiter { open: span, close: span, kind },
469                        tt::TokenTreesView::new(&[]),
470                    )),
471                    SyntaxFixupUndoInfo::default(),
472                    span,
473                )
474            };
475
476            let Some(tt) = node.token_tree() else {
477                return dummy_tt(tt::DelimiterKind::Invisible);
478            };
479            let first = tt.left_delimiter_token().map(|it| it.kind()).unwrap_or(T!['(']);
480            let last = tt.right_delimiter_token().map(|it| it.kind()).unwrap_or(T![.]);
481
482            let mismatched_delimiters = !matches!(
483                (first, last),
484                (T!['('], T![')']) | (T!['['], T![']']) | (T!['{'], T!['}'])
485            );
486            if mismatched_delimiters {
487                // Don't expand malformed (unbalanced) macro invocations. This is
488                // less than ideal, but trying to expand unbalanced  macro calls
489                // sometimes produces pathological, deeply nested code which breaks
490                // all kinds of things.
491                //
492                // So instead, we'll return an empty subtree here
493                cov_mark::hit!(issue9358_bad_macro_stack_overflow);
494
495                let kind = match first {
496                    _ if loc.def.is_proc_macro() => tt::DelimiterKind::Invisible,
497                    T!['('] => tt::DelimiterKind::Parenthesis,
498                    T!['['] => tt::DelimiterKind::Bracket,
499                    T!['{'] => tt::DelimiterKind::Brace,
500                    _ => tt::DelimiterKind::Invisible,
501                };
502                return dummy_tt(kind);
503            }
504
505            let mut tt = syntax_bridge::syntax_node_to_token_tree(
506                tt.syntax(),
507                map.as_ref(),
508                span,
509                if loc.def.is_proc_macro() {
510                    DocCommentDesugarMode::ProcMacro
511                } else {
512                    DocCommentDesugarMode::Mbe
513                },
514            );
515            if loc.def.is_proc_macro() {
516                // proc macros expect their inputs without parentheses, MBEs expect it with them included
517                tt.top_subtree_delimiter_mut().kind = tt::DelimiterKind::Invisible;
518            }
519            return (Arc::new(tt), SyntaxFixupUndoInfo::NONE, span);
520        }
521        // MacroCallKind::Derive should not be here. As we are getting the argument for the derive macro
522        MacroCallKind::Derive { .. } => {
523            unreachable!("`ExpandDatabase::macro_arg` called with `MacroCallKind::Derive`")
524        }
525        MacroCallKind::Attr { ast_id, invoc_attr_index, .. } => {
526            let node = ast_id.to_ptr(db).to_node(&root);
527            let attr_source = attr_source(invoc_attr_index, &node);
528
529            let span = map.span_for_range(
530                attr_source
531                    .as_ref()
532                    .and_then(|it| it.path())
533                    .map_or_else(|| node.syntax().text_range(), |it| it.syntax().text_range()),
534            );
535            // If derive attribute we need to censor the derive input
536            if matches!(loc.def.kind, MacroDefKind::BuiltInAttr(_, expander) if expander.is_derive())
537                && ast::Adt::can_cast(node.syntax().kind())
538            {
539                let adt = ast::Adt::cast(node.syntax().clone()).unwrap();
540                let censor_derive_input = censor_derive_input(invoc_attr_index, &adt);
541                (censor_derive_input, node, span)
542            } else {
543                (attr_source.into_iter().map(|it| it.syntax().clone().into()).collect(), node, span)
544            }
545        }
546    };
547
548    let (mut tt, undo_info) = {
549        let syntax = item_node.syntax();
550        let censor_cfg = cfg_process::process_cfg_attrs(db, syntax, &loc).unwrap_or_default();
551        let mut fixups =
552            fixup::fixup_syntax(map.as_ref(), syntax, span, DocCommentDesugarMode::ProcMacro);
553        fixups.append.retain(|it, _| match it {
554            syntax::NodeOrToken::Token(_) => true,
555            it => !censor.contains(it) && !censor_cfg.contains(it),
556        });
557        fixups.remove.extend(censor);
558        fixups.remove.extend(censor_cfg);
559
560        (
561            syntax_bridge::syntax_node_to_token_tree_modified(
562                syntax,
563                map,
564                fixups.append,
565                fixups.remove,
566                span,
567                DocCommentDesugarMode::ProcMacro,
568            ),
569            fixups.undo_info,
570        )
571    };
572
573    if loc.def.is_proc_macro() {
574        // proc macros expect their inputs without parentheses, MBEs expect it with them included
575        tt.top_subtree_delimiter_mut().kind = tt::DelimiterKind::Invisible;
576    }
577
578    (Arc::new(tt), undo_info, span)
579}

581// FIXME: Censoring info should be calculated by the caller! Namely by name resolution
582/// Derives expect all `#[derive(..)]` invocations up to (and including) the currently invoked one to be stripped
583fn censor_derive_input(derive_attr_index: AttrId, node: &ast::Adt) -> FxHashSet<SyntaxElement> {
584    // FIXME: handle `cfg_attr`
585    cov_mark::hit!(derive_censoring);
586    collect_attrs(node)
587        .take(derive_attr_index.ast_index() + 1)
588        .filter_map(|(_, attr)| Either::left(attr))
589        // FIXME, this resolution should not be done syntactically
590        // derive is a proper macro now, no longer builtin
591        // But we do not have resolution at this stage, this means
592        // we need to know about all macro calls for the given ast item here
593        // so we require some kind of mapping...
594        .filter(|attr| attr.simple_name().as_deref() == Some("derive"))
595        .map(|it| it.syntax().clone().into())
596        .collect()
597}

599/// Attributes expect the invoking attribute to be stripped
600fn attr_source(invoc_attr_index: AttrId, node: &ast::Item) -> Option<ast::Attr> {
601    // FIXME: handle `cfg_attr`
602    cov_mark::hit!(attribute_macro_attr_censoring);
603    collect_attrs(node).nth(invoc_attr_index.ast_index()).and_then(|(_, attr)| Either::left(attr))
604}

606impl TokenExpander {
607    fn macro_expander(db: &dyn ExpandDatabase, id: MacroDefId) -> TokenExpander {
608        match id.kind {
609            MacroDefKind::Declarative(ast_id) => {
610                TokenExpander::DeclarativeMacro(db.decl_macro_expander(id.krate, ast_id))
611            }
612            MacroDefKind::BuiltIn(_, expander) => TokenExpander::BuiltIn(expander),
613            MacroDefKind::BuiltInAttr(_, expander) => TokenExpander::BuiltInAttr(expander),
614            MacroDefKind::BuiltInDerive(_, expander) => TokenExpander::BuiltInDerive(expander),
615            MacroDefKind::BuiltInEager(_, expander) => TokenExpander::BuiltInEager(expander),
616            MacroDefKind::ProcMacro(_, expander, _) => TokenExpander::ProcMacro(expander),
617        }
618    }
619}
620
621enum CowArc<T> {
622    Arc(Arc<T>),
623    Owned(T),
624}
625
626fn macro_expand(
627    db: &dyn ExpandDatabase,
628    macro_call_id: MacroCallId,
629    loc: MacroCallLoc,
630) -> ExpandResult<(CowArc<tt::TopSubtree>, MatchedArmIndex)> {
631    let _p = tracing::info_span!("macro_expand").entered();
632
633    let (ExpandResult { value: (tt, matched_arm), err }, span) = match loc.def.kind {
634        MacroDefKind::ProcMacro(..) => {
635            return db.expand_proc_macro(macro_call_id).map(CowArc::Arc).zip_val(None);
636        }
637        _ => {
638            let (macro_arg, undo_info, span) =
639                db.macro_arg_considering_derives(macro_call_id, &loc.kind);
640
641            let arg = &*macro_arg;
642            let res =
643                match loc.def.kind {
644                    MacroDefKind::Declarative(id) => db
645                        .decl_macro_expander(loc.def.krate, id)
646                        .expand(db, arg.clone(), macro_call_id, span),
647                    MacroDefKind::BuiltIn(_, it) => {
648                        it.expand(db, macro_call_id, arg, span).map_err(Into::into).zip_val(None)
649                    }
650                    MacroDefKind::BuiltInDerive(_, it) => {
651                        it.expand(db, macro_call_id, arg, span).map_err(Into::into).zip_val(None)
652                    }
653                    MacroDefKind::BuiltInEager(_, it) => {
654                        // This might look a bit odd, but we do not expand the inputs to eager macros here.
655                        // Eager macros inputs are expanded, well, eagerly when we collect the macro calls.
656                        // That kind of expansion uses the ast id map of an eager macros input though which goes through
657                        // the HirFileId machinery. As eager macro inputs are assigned a macro file id that query
658                        // will end up going through here again, whereas we want to just want to inspect the raw input.
659                        // As such we just return the input subtree here.
660                        let eager = match &loc.kind {
661                            MacroCallKind::FnLike { eager: None, .. } => {
662                                return ExpandResult::ok(CowArc::Arc(macro_arg.clone()))
663                                    .zip_val(None);
664                            }
665                            MacroCallKind::FnLike { eager: Some(eager), .. } => Some(&**eager),
666                            _ => None,
667                        };
668
669                        let mut res = it.expand(db, macro_call_id, arg, span).map_err(Into::into);
670
671                        if let Some(EagerCallInfo { error, .. }) = eager {
672                            // FIXME: We should report both errors!
673                            res.err = error.clone().or(res.err);
674                        }
675                        res.zip_val(None)
676                    }
677                    MacroDefKind::BuiltInAttr(_, it) => {
678                        let mut res = it.expand(db, macro_call_id, arg, span);
679                        fixup::reverse_fixups(&mut res.value, &undo_info);
680                        res.zip_val(None)
681                    }
682                    MacroDefKind::ProcMacro(_, _, _) => unreachable!(),
683                };
684            (ExpandResult { value: res.value, err: res.err }, span)
685        }
686    };
687
688    // Skip checking token tree limit for include! macro call
689    if !loc.def.is_include() {
690        // Set a hard limit for the expanded tt
691        if let Err(value) = check_tt_count(&tt) {
692            return value
693                .map(|()| CowArc::Owned(tt::TopSubtree::empty(tt::DelimSpan::from_single(span))))
694                .zip_val(matched_arm);
695        }
696    }
697
698    ExpandResult { value: (CowArc::Owned(tt), matched_arm), err }
699}

701fn proc_macro_span(db: &dyn ExpandDatabase, ast: AstId<ast::Fn>) -> Span {
702    let root = db.parse_or_expand(ast.file_id);
703    let ast_id_map = &db.ast_id_map(ast.file_id);
704    let span_map = &db.span_map(ast.file_id);
705
706    let node = ast_id_map.get(ast.value).to_node(&root);
707    let range = ast::HasName::name(&node)
708        .map_or_else(|| node.syntax().text_range(), |name| name.syntax().text_range());
709    span_map.span_for_range(range)
710}
711
712fn expand_proc_macro(
713    db: &dyn ExpandDatabase,
714    id: MacroCallId,
715) -> ExpandResult<Arc<tt::TopSubtree>> {
716    let loc = db.lookup_intern_macro_call(id);
717    let (macro_arg, undo_info, span) = db.macro_arg_considering_derives(id, &loc.kind);
718
719    let (ast, expander) = match loc.def.kind {
720        MacroDefKind::ProcMacro(ast, expander, _) => (ast, expander),
721        _ => unreachable!(),
722    };
723
724    let attr_arg = match &loc.kind {
725        MacroCallKind::Attr { attr_args: Some(attr_args), .. } => Some(&**attr_args),
726        _ => None,
727    };
728
729    let ExpandResult { value: mut tt, err } = {
730        let span = db.proc_macro_span(ast);
731        expander.expand(
732            db,
733            loc.def.krate,
734            loc.krate,
735            &macro_arg,
736            attr_arg,
737            span_with_def_site_ctxt(db, span, id, loc.def.edition),
738            span_with_call_site_ctxt(db, span, id, loc.def.edition),
739            span_with_mixed_site_ctxt(db, span, id, loc.def.edition),
740        )
741    };
742
743    // Set a hard limit for the expanded tt
744    if let Err(value) = check_tt_count(&tt) {
745        return value.map(|()| Arc::new(tt::TopSubtree::empty(tt::DelimSpan::from_single(span))));
746    }
747
748    fixup::reverse_fixups(&mut tt, &undo_info);
749
750    ExpandResult { value: Arc::new(tt), err }
751}

753pub(crate) fn token_tree_to_syntax_node(
754    db: &dyn ExpandDatabase,
755    tt: &tt::TopSubtree,
756    expand_to: ExpandTo,
757    edition: parser::Edition,
758) -> (Parse<SyntaxNode>, ExpansionSpanMap) {
759    let entry_point = match expand_to {
760        ExpandTo::Statements => syntax_bridge::TopEntryPoint::MacroStmts,
761        ExpandTo::Items => syntax_bridge::TopEntryPoint::MacroItems,
762        ExpandTo::Pattern => syntax_bridge::TopEntryPoint::Pattern,
763        ExpandTo::Type => syntax_bridge::TopEntryPoint::Type,
764        ExpandTo::Expr => syntax_bridge::TopEntryPoint::Expr,
765    };
766    syntax_bridge::token_tree_to_syntax_node(tt, entry_point, &mut |ctx| ctx.edition(db), edition)
767}
768
769fn check_tt_count(tt: &tt::TopSubtree) -> Result<(), ExpandResult<()>> {
770    let tt = tt.top_subtree();
771    let count = tt.count();
772    if count <= TOKEN_LIMIT {
773        Ok(())
774    } else {
775        Err(ExpandResult {
776            value: (),
777            err: Some(ExpandError::other(
778                tt.delimiter.open,
779                format!(
780                    "macro invocation exceeds token limit: produced {} tokens, limit is {}",
781                    count, TOKEN_LIMIT,
782                ),
783            )),
784        })
785    }
786}
