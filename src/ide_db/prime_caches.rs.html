1//! rust-analyzer is lazy and doesn't compute anything unless asked. This
2//! sometimes is counter productive when, for example, the first goto definition
3//! request takes longer to compute. This module implements prepopulation of
4//! various caches, it's not really advanced at the moment.
5mod topologic_sort;
6

9use hir::{Symbol, db::DefDatabase};
10use itertools::Itertools;
11use salsa::{Cancelled, Database};
12
13use crate::{
14    FxIndexMap, RootDatabase,
15    base_db::{Crate, RootQueryDb},
16    symbol_index::SymbolsDatabase,
17};

19/// We're indexing many crates.
20#[derive(Debug)]
21pub struct ParallelPrimeCachesProgress {
22    /// the crates that we are currently priming.
23    pub crates_currently_indexing: Vec<Symbol>,
24    /// the total number of crates we want to prime.
25    pub crates_total: usize,
26    /// the total number of crates that have finished priming
27    pub crates_done: usize,
28    pub work_type: &'static str,
29}
30
31pub fn parallel_prime_caches(
32    db: &RootDatabase,
33    num_worker_threads: usize,
34    cb: &(dyn Fn(ParallelPrimeCachesProgress) + Sync),
35) {
36    let _p = tracing::info_span!("parallel_prime_caches").entered();
37
38    let mut crates_to_prime = {
39        // FIXME: We already have the crate list topologically sorted (but without the things
40        // `TopologicalSortIter` gives us). Maybe there is a way to avoid using it and rip it out
41        // of the codebase?
42        let mut builder = topologic_sort::TopologicalSortIter::builder();
43
44        for &crate_id in db.all_crates().iter() {
45            builder.add(crate_id, crate_id.data(db).dependencies.iter().map(|d| d.crate_id));
46        }
47
48        builder.build()
49    };
50
51    enum ParallelPrimeCacheWorkerProgress {
52        BeginCrate { crate_id: Crate, crate_name: Symbol },
53        EndCrate { crate_id: Crate },
54    }
55
56    // We split off def map computation from other work,
57    // as the def map is the relevant one. Once the defmaps are computed
58    // the project is ready to go, the other indices are just nice to have for some IDE features.
59    #[derive(PartialOrd, Ord, PartialEq, Eq, Copy, Clone)]
60    enum PrimingPhase {
61        DefMap,
62        ImportMap,
63        CrateSymbols,
64    }
65
66    let (work_sender, progress_receiver) = {
67        let (progress_sender, progress_receiver) = crossbeam_channel::unbounded();
68        let (work_sender, work_receiver) = crossbeam_channel::unbounded();
69        let prime_caches_worker = move |db: RootDatabase| {
70            while let Ok((crate_id, crate_name, kind)) = work_receiver.recv() {
71                progress_sender
72                    .send(ParallelPrimeCacheWorkerProgress::BeginCrate { crate_id, crate_name })?;
73
74                match kind {
75                    PrimingPhase::DefMap => _ = db.crate_def_map(crate_id),
76                    PrimingPhase::ImportMap => _ = db.import_map(crate_id),
77                    PrimingPhase::CrateSymbols => _ = db.crate_symbols(crate_id.into()),
78                }
79
80                progress_sender.send(ParallelPrimeCacheWorkerProgress::EndCrate { crate_id })?;
81            }
82
83            Ok::<_, crossbeam_channel::SendError<_>>(())
84        };
85
86        for id in 0..num_worker_threads {
87            let worker = prime_caches_worker.clone();
88            let db = db.snapshot();
89
90            stdx::thread::Builder::new(stdx::thread::ThreadIntent::Worker)
91                .allow_leak(true)
92                .name(format!("PrimeCaches#{id}"))
93                .spawn(move || Cancelled::catch(|| worker(db.snapshot())))
94                .expect("failed to spawn thread");
95        }
96
97        (work_sender, progress_receiver)
98    };
99
100    let crates_total = crates_to_prime.pending();
101    let mut crates_done = 0;
102
103    // an index map is used to preserve ordering so we can sort the progress report in order of
104    // "longest crate to index" first
105    let mut crates_currently_indexing =
106        FxIndexMap::with_capacity_and_hasher(num_worker_threads, Default::default());
107
108    let mut additional_phases = vec![];
109
110    while crates_done < crates_total {
111        db.unwind_if_revision_cancelled();
112
113        for krate in &mut crates_to_prime {
114            let name = krate.extra_data(db).display_name.as_deref().cloned().unwrap_or_else(|| {
115                Symbol::integer(salsa::plumbing::AsId::as_id(&krate).as_u32() as usize)
116            });
117            let origin = &krate.data(db).origin;
118            if origin.is_lang() {
119                additional_phases.push((krate, name.clone(), PrimingPhase::ImportMap));
120            } else if origin.is_local() {
121                // Compute the symbol search index.
122                // This primes the cache for `ide_db::symbol_index::world_symbols()`.
123                //
124                // We do this for workspace crates only (members of local_roots), because doing it
125                // for all dependencies could be *very* unnecessarily slow in a large project.
126                //
127                // FIXME: We should do it unconditionally if the configuration is set to default to
128                // searching dependencies (rust-analyzer.workspace.symbol.search.scope), but we
129                // would need to pipe that configuration information down here.
130                additional_phases.push((krate, name.clone(), PrimingPhase::CrateSymbols));
131            }
132
133            work_sender.send((krate, name, PrimingPhase::DefMap)).ok();
134        }
135
136        // recv_timeout is somewhat a hack, we need a way to from this thread check to see if the current salsa revision
137        // is cancelled on a regular basis. workers will only exit if they are processing a task that is cancelled, or
138        // if this thread exits, and closes the work channel.
139        let worker_progress = match progress_receiver.recv_timeout(Duration::from_millis(10)) {
140            Ok(p) => p,
141            Err(crossbeam_channel::RecvTimeoutError::Timeout) => {
142                continue;
143            }
144            Err(crossbeam_channel::RecvTimeoutError::Disconnected) => {
145                // our workers may have died from a cancelled task, so we'll check and re-raise here.
146                db.unwind_if_revision_cancelled();
147                break;
148            }
149        };
150        match worker_progress {
151            ParallelPrimeCacheWorkerProgress::BeginCrate { crate_id, crate_name } => {
152                crates_currently_indexing.insert(crate_id, crate_name);
153            }
154            ParallelPrimeCacheWorkerProgress::EndCrate { crate_id } => {
155                crates_currently_indexing.swap_remove(&crate_id);
156                crates_to_prime.mark_done(crate_id);
157                crates_done += 1;
158            }
159        };
160
161        let progress = ParallelPrimeCachesProgress {
162            crates_currently_indexing: crates_currently_indexing.values().cloned().collect(),
163            crates_done,
164            crates_total,
165            work_type: "Indexing",
166        };
167
168        cb(progress);
169    }
170
171    let mut crates_done = 0;
172    let crates_total = additional_phases.len();
173    for w in additional_phases.into_iter().sorted_by_key(|&(_, _, phase)| phase) {
174        work_sender.send(w).ok();
175    }
176
177    while crates_done < crates_total {
178        db.unwind_if_revision_cancelled();
179
180        // recv_timeout is somewhat a hack, we need a way to from this thread check to see if the current salsa revision
181        // is cancelled on a regular basis. workers will only exit if they are processing a task that is cancelled, or
182        // if this thread exits, and closes the work channel.
183        let worker_progress = match progress_receiver.recv_timeout(Duration::from_millis(10)) {
184            Ok(p) => p,
185            Err(crossbeam_channel::RecvTimeoutError::Timeout) => {
186                continue;
187            }
188            Err(crossbeam_channel::RecvTimeoutError::Disconnected) => {
189                // our workers may have died from a cancelled task, so we'll check and re-raise here.
190                db.unwind_if_revision_cancelled();
191                break;
192            }
193        };
194        match worker_progress {
195            ParallelPrimeCacheWorkerProgress::BeginCrate { crate_id, crate_name } => {
196                crates_currently_indexing.insert(crate_id, crate_name);
197            }
198            ParallelPrimeCacheWorkerProgress::EndCrate { crate_id } => {
199                crates_currently_indexing.swap_remove(&crate_id);
200                crates_done += 1;
201            }
202        };
203
204        let progress = ParallelPrimeCachesProgress {
205            crates_currently_indexing: crates_currently_indexing.values().cloned().collect(),
206            crates_done,
207            crates_total,
208            work_type: "Populating symbols",
209        };
210
211        cb(progress);
212    }
