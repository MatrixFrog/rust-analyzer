1//! Runs `rustc --print target-spec-json` to get the target_data_layout.
2
3use anyhow::Context;
4use rustc_hash::FxHashMap;
5use toolchain::Tool;
6
7use crate::{Sysroot, toolchain_info::QueryConfig, utf8_stdout};
8
9/// Uses `rustc --print target-spec-json`.
10pub fn get(
11    config: QueryConfig<'_>,
12    target: Option<&str>,
13    extra_env: &FxHashMap<String, String>,
14) -> anyhow::Result<String> {
15    const RUSTC_ARGS: [&str; 2] = ["--print", "target-spec-json"];
16    let process = |output: String| {
17        (|| Some(output.split_once(r#""data-layout": ""#)?.1.split_once('"')?.0.to_owned()))()
18            .ok_or_else(|| {
19                anyhow::format_err!("could not parse target-spec-json from command output")
20            })
21    };
22    let (sysroot, current_dir) = match config {
23        QueryConfig::Cargo(sysroot, cargo_toml) => {
24            let mut cmd = sysroot.tool(Tool::Cargo, cargo_toml.parent());
25            cmd.envs(extra_env);
26            cmd.env("RUSTC_BOOTSTRAP", "1");
27            cmd.args(["rustc", "-Z", "unstable-options"]).args(RUSTC_ARGS).args([
28                "--",
29                "-Z",
30                "unstable-options",
31            ]);
32            if let Some(target) = target {
33                cmd.args(["--target", target]);
34            }
35            match utf8_stdout(&mut cmd) {
36                Ok(output) => return process(output),
37                Err(e) => {
38                    tracing::warn!(%e, "failed to run `{cmd:?}`, falling back to invoking rustc directly");
39                    (sysroot, cargo_toml.parent().as_ref())
40                }
41            }
42        }
43        QueryConfig::Rustc(sysroot, current_dir) => (sysroot, current_dir),
44    };
45
46    let mut cmd = Sysroot::tool(sysroot, Tool::Rustc, current_dir);
47    cmd.envs(extra_env)
48        .env("RUSTC_BOOTSTRAP", "1")
49        .args(["-Z", "unstable-options"])
50        .args(RUSTC_ARGS);
51    if let Some(target) = target {
52        cmd.args(["--target", target]);
53    }
54    utf8_stdout(&mut cmd)
55        .with_context(|| format!("unable to fetch target-data-layout via `{cmd:?}`"))
56        .and_then(process)
57}

59#[cfg(test)]
60mod tests {
61    use paths::{AbsPathBuf, Utf8PathBuf};
62
63    use crate::{ManifestPath, Sysroot};
64
65    use super::*;
66
67    #[test]
68    fn cargo() {
69        let manifest_path = concat!(env!("CARGO_MANIFEST_DIR"), "/Cargo.toml");
70        let sysroot = Sysroot::empty();
71        let manifest_path =
72            ManifestPath::try_from(AbsPathBuf::assert(Utf8PathBuf::from(manifest_path))).unwrap();
73        let cfg = QueryConfig::Cargo(&sysroot, &manifest_path);
74        assert!(get(cfg, None, &FxHashMap::default()).is_ok());
75    }
76
77    #[test]
78    fn rustc() {
79        let sysroot = Sysroot::empty();
80        let cfg = QueryConfig::Rustc(&sysroot, env!("CARGO_MANIFEST_DIR").as_ref());
81        assert!(get(cfg, None, &FxHashMap::default()).is_ok());
82    }
